024package org.metastatic.rsync;
025
026import java.io.IOException;
027import java.io.InputStream;
028
029import java.lang.reflect.Constructor;
030import java.lang.reflect.InvocationTargetException;
031
032import java.util.List;
033
034/**
035 * The base class of objects that decode (internalize) checksum pairs
036 * from byte streams.
037 *
038 * @version $Revision$
039 */
040public abstract class ChecksumDecoder
041{
042
043    // Constants and fields.
044    // -------------------------------------------------------------------------
045
046    /**
047     * Property prefix for checksum encoders.
048     */
049    public static final String PROPERTY = "jarsync.checksumDecoder.";
050
051    /**
052     * The configuration object.
053     */
054    protected Configuration config;
055
056    /**
057     * The input stream being read from.
058     */
059    protected InputStream in;
060
061    // Constructor.
062    // -------------------------------------------------------------------------
063
064    public ChecksumDecoder(Configuration config, InputStream in)
065    {
066        this.config = (Configuration) config.clone();
067        this.in = in;
068    }
069
070    // Class method.
071    // -------------------------------------------------------------------------
072
073    /**
074     * Gets an instance of a checksum decoder for the specified
075     * encoding.
076     *
077     * @param encoding The encoding name.
078     * @param config   The configuration object.
079     * @param in       The input stream.
080     * @throws NullPointerException     If any parameter is null.
081     * @throws IllegalArgumentException If the specified encoding cannot
082     *                                  be found, or if any of the arguments are inappropriate.
083     */
084    public static ChecksumEncoder getInstance(String encoding,
085                                              Configuration config,
086                                              InputStream in)
087    {
088        if (encoding == null || config == null || in == null)
089            throw new NullPointerException();
090        if (encoding.length() == 0)
091            throw new IllegalArgumentException();
092        try
093        {
094            Class clazz = Class.forName(System.getProperty(PROPERTY + encoding));
095            if (!ChecksumEncoder.class.isAssignableFrom(clazz))
096                throw new IllegalArgumentException(clazz.getName() +
097                        ": not a subclass of " +
098                        ChecksumEncoder.class.getName());
099            Constructor c = clazz.getConstructor(new Class[]{Configuration.class,
100                    InputStream.class});
101            return (ChecksumEncoder) c.newInstance(new Object[]{config, in});
102        } catch (ClassNotFoundException cnfe)
103        {
104            throw new IllegalArgumentException("class not found: " +
105                    cnfe.getMessage());
106        } catch (NoSuchMethodException nsme)
107        {
108            throw new IllegalArgumentException("subclass has no constructor");
109        } catch (InvocationTargetException ite)
110        {
111            throw new IllegalArgumentException(ite.getMessage());
112        } catch (InstantiationException ie)
113        {
114            throw new IllegalArgumentException(ie.getMessage());
115        } catch (IllegalAccessException iae)
116        {
117            throw new IllegalArgumentException(iae.getMessage());
118        }
119    }
120
121    // Instance methods.
122    // -------------------------------------------------------------------------
123
124    /**
125     * Decodes checksums from the stream, storing them into the
126     * specified list, until the end of checksums is encountered.
127     *
128     * @param sums The list to store the sums into.
129     * @return The number of checksums read.
130     * @throws IOException          If an I/O error occurs.
131     * @throws NullPointerException If any element of the list is null.
132     */
133    public int read(List<ChecksumPair> sums) throws IOException
134    {
135        if (sums == null)
136            throw new NullPointerException();
137        int count = 0;
138        ChecksumPair pair;
139        while ((pair = read()) != null)
140        {
141            sums.add(pair);
142            ++count;
143        }
144        return count;
145    }
146
147    // Abstract methods.
148    // -------------------------------------------------------------------------
149
150    /**
151     * Decodes a checksum pair from the input stream.
152     *
153     * @return The pair read, or null if the end of stream is encountered.
154     * @throws IOException If an I/O error occurs.
155     */
156    public abstract ChecksumPair read() throws IOException;
157}

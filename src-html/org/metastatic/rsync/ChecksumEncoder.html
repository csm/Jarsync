024package org.metastatic.rsync;
025
026import java.io.IOException;
027import java.io.OutputStream;
028import java.lang.reflect.Constructor;
029import java.lang.reflect.InvocationTargetException;
030import java.util.List;
031
032/**
033 * The base class of objects that encode (externalize) checksum pairs
034 * to byte streams.
035 *
036 * @version $Revision$
037 */
038public abstract class ChecksumEncoder
039{
040
041    // Constants and fields.
042    // -------------------------------------------------------------------------
043
044    /**
045     * Property prefix for checksum encoders.
046     */
047    public static final String PROPERTY = "jarsync.checksumEncoder.";
048
049    /**
050     * The configuration object.
051     */
052    protected Configuration config;
053
054    /**
055     * The output stream being written to.
056     */
057    protected OutputStream out;
058
059    // Constructor.
060    // -------------------------------------------------------------------------
061
062    public ChecksumEncoder(Configuration config, OutputStream out)
063    {
064        this.config = (Configuration) config.clone();
065        this.out = out;
066    }
067
068    // Class method.
069    // -------------------------------------------------------------------------
070
071    /**
072     * Gets an instance of a checksum encoder for the specified
073     * encoding.
074     *
075     * @param encoding The encoding name.
076     * @param config   The configuration object.
077     * @param out      The output stream.
078     * @throws NullPointerException     If any parameter is null.
079     * @throws IllegalArgumentException If the specified encoding cannot
080     *                                  be found, or if any of the arguments are inappropriate.
081     */
082    public static ChecksumEncoder getInstance(String encoding,
083                                              Configuration config,
084                                              OutputStream out)
085    {
086        if (encoding == null || config == null || out == null)
087            throw new NullPointerException();
088        if (encoding.length() == 0)
089            throw new IllegalArgumentException();
090        try
091        {
092            Class clazz = Class.forName(System.getProperty(PROPERTY + encoding));
093            if (!ChecksumEncoder.class.isAssignableFrom(clazz))
094                throw new IllegalArgumentException(clazz.getName() +
095                        ": not a subclass of " +
096                        ChecksumEncoder.class.getName());
097            Constructor c = clazz.getConstructor(new Class[]{Configuration.class,
098                    OutputStream.class});
099            return (ChecksumEncoder) c.newInstance(new Object[]{config, out});
100        } catch (ClassNotFoundException cnfe)
101        {
102            throw new IllegalArgumentException("class not found: " +
103                    cnfe.getMessage());
104        } catch (NoSuchMethodException nsme)
105        {
106            throw new IllegalArgumentException("subclass has no constructor");
107        } catch (InvocationTargetException ite)
108        {
109            throw new IllegalArgumentException(ite.getMessage());
110        } catch (InstantiationException ie)
111        {
112            throw new IllegalArgumentException(ie.getMessage());
113        } catch (IllegalAccessException iae)
114        {
115            throw new IllegalArgumentException(iae.getMessage());
116        }
117    }
118
119    // Instance methods.
120    // -------------------------------------------------------------------------
121
122    /**
123     * Encodes a list of checksums to the output stream.
124     *
125     * @param sums The sums to write.
126     * @throws IOException              If an I/O error occurs.
127     * @throws NullPointerException     If any element of the list is null.
128     * @throws IllegalArgumentException If any element of the list is
129     *                                  not a {@link ChecksumPair}.
130     */
131    public void write(List<ChecksumPair> sums) throws IOException
132    {
133        for (ChecksumPair o : sums)
134        {
135            if (o == null)
136                throw new NullPointerException();
137        }
138        for (ChecksumPair sum : sums)
139        {
140            write(sum);
141        }
142    }
143
144    // Abstract methods.
145    // -------------------------------------------------------------------------
146
147    /**
148     * Encodes a checksum pair to the output stream.
149     *
150     * @param pair The pair to write.
151     * @throws IOException If an I/O error occurs.
152     */
153    public abstract void write(ChecksumPair pair) throws IOException;
154
155    /**
156     * Finishes encoding by emitting any end-of-checksums markers.
157     *
158     * @throws IOException If an I/O error occurs.
159     */
160    public abstract void doFinal() throws IOException;
161
162    /**
163     * This method returns <code>true</code> if the checksums must be
164     * presented in order of ascending offset.
165     *
166     * @return true if this encoder requires order.
167     */
168    public abstract boolean requiresOrder();
169}

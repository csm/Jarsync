044package org.metastatic.rsync;
045
046/**
047 * A general interface for 32-bit checksums that have the "rolling"
048 * property.
049 *
050 * @author Casey Marshall
051 * @version $Revision$
052 */
053public interface RollingChecksum extends Cloneable, java.io.Serializable
054{
055
056    // Methods.
057    // -----------------------------------------------------------------------
058
059    /**
060     * Returns the currently-computed 32-bit checksum.
061     *
062     * @return The checksum.
063     */
064    int getValue();
065
066    /**
067     * Resets the internal state of the checksum, so it may be re-used
068     * later.
069     */
070    void reset();
071
072    /**
073     * Update the checksum with a single byte. This is where the
074     * "rolling" method is used.
075     *
076     * @param bt The next byte.
077     */
078    void roll(byte bt);
079
080    /**
081     * Update the checksum by simply "trimming" the
082     * least-recently-updated byte from the internal state. Most, but not
083     * all, checksums can support this.
084     */
085    void trim();
086
087    /**
088     * Replaces the current internal state with entirely new data.
089     *
090     * @param buf    The bytes to checksum.
091     * @param offset The offset into <code>buf</code> to start reading.
092     * @param length The number of bytes to update.
093     */
094    void check(byte[] buf, int offset, int length);
095
096    /**
097     * Copies this checksum instance into a new instance. This method
098     * should be optional, and only implemented if the class implements
099     * the {@link java.lang.Cloneable} interface.
100     *
101     * @return A clone of this instance.
102     */
103    Object clone();
104
105    /**
106     * Tests if a particular checksum is equal to this checksum. This
107     * means that the other object is an instance of this class, and its
108     * internal state equals this checksum's internal state.
109     *
110     * @param o The object to test.
111     * @return <code>true</code> if this checksum equals the other
112     * checksum.
113     */
114    boolean equals(Object o);
115}

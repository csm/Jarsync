023package org.metastatic.rsync;
024
025/**
026 * A general interface for 32-bit checksums that have the "rolling"
027 * property.
028 *
029 * @author Casey Marshall
030 * @version $Revision$
031 */
032public interface RollingChecksum extends Cloneable, java.io.Serializable
033{
034
035    // Methods.
036    // -----------------------------------------------------------------------
037
038    /**
039     * Returns the currently-computed 32-bit checksum.
040     *
041     * @return The checksum.
042     */
043    int getValue();
044
045    /**
046     * Resets the internal state of the checksum, so it may be re-used
047     * later.
048     */
049    void reset();
050
051    /**
052     * Update the checksum with a single byte. This is where the
053     * "rolling" method is used.
054     *
055     * @param bt The next byte.
056     */
057    void roll(byte bt);
058
059    /**
060     * Update the checksum by simply "trimming" the
061     * least-recently-updated byte from the internal state. Most, but not
062     * all, checksums can support this.
063     */
064    void trim();
065
066    /**
067     * Replaces the current internal state with entirely new data.
068     *
069     * @param buf    The bytes to checksum.
070     * @param offset The offset into <code>buf</code> to start reading.
071     * @param length The number of bytes to update.
072     */
073    void check(byte[] buf, int offset, int length);
074
075    /**
076     * Copies this checksum instance into a new instance. This method
077     * should be optional, and only implemented if the class implements
078     * the {@link java.lang.Cloneable} interface.
079     *
080     * @return A clone of this instance.
081     */
082    Object clone();
083
084    /**
085     * Tests if a particular checksum is equal to this checksum. This
086     * means that the other object is an instance of this class, and its
087     * internal state equals this checksum's internal state.
088     *
089     * @param o The object to test.
090     * @return <code>true</code> if this checksum equals the other
091     * checksum.
092     */
093    boolean equals(Object o);
094}

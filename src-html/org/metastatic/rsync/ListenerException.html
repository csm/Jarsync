044package org.metastatic.rsync;
045
046/**
047 * Signals an exception raised by an {@link org.metastatic.rsync.GeneratorListener}, {@link
048 * org.metastatic.rsync.MatcherListener}, or {@link org.metastatic.rsync.RebuilderListener}.
049 *
050 * <p>Listener exceptions may contain other exceptions (the "cause") and
051 * may be chained together if there are multiple failures accross
052 * multiple listeners.
053 */
054public class ListenerException extends Exception
055{
056
057    // Fields.
058    // -----------------------------------------------------------------------
059
060    protected ListenerException next;
061
062    protected Throwable cause;
063
064    // Constructors.
065    // -----------------------------------------------------------------------
066
067    public ListenerException(Throwable cause)
068    {
069        super();
070        this.cause = cause;
071    }
072
073    public ListenerException(Throwable cause, String msg)
074    {
075        super(msg);
076        this.cause = cause;
077    }
078
079    public ListenerException(String msg)
080    {
081        super(msg);
082    }
083
084    public ListenerException()
085    {
086        super();
087    }
088
089    // Instance methods.
090    // -----------------------------------------------------------------------
091
092    /**
093     * Returns the next exception in this chain, or <code>null</code> if
094     * there are no other exceptions.
095     *
096     * @return The next exception.
097     */
098    public ListenerException getNext()
099    {
100        return next;
101    }
102
103    /**
104     * Sets the next exception in this chain.
105     *
106     * @param next The next exception.
107     */
108    public void setNext(ListenerException next)
109    {
110        this.next = next;
111    }
112
113    /**
114     * Gets the cause of this exception, or <code>null</code> if the
115     * cause is unknown.
116     *
117     * @return The cause.
118     */
119    public Throwable getCause()
120    {
121        return cause;
122    }
123
124    /**
125     * Sets the cause of this exception.
126     *
127     * @param cause The cause of this exception.
128     */
129    public synchronized Throwable initCause(Throwable cause)
130    {
131        Throwable old = this.cause;
132        this.cause = cause;
133        return old;
134    }
135}

024package org.metastatic.rsync;
025
026/**
027 * Signals an exception raised by an {@link org.metastatic.rsync.GeneratorListener}, {@link
028 * org.metastatic.rsync.MatcherListener}, or {@link org.metastatic.rsync.RebuilderListener}.
029 *
030 * <p>Listener exceptions may contain other exceptions (the "cause") and
031 * may be chained together if there are multiple failures accross
032 * multiple listeners.
033 */
034public class ListenerException extends Exception
035{
036
037    // Fields.
038    // -----------------------------------------------------------------------
039
040    protected ListenerException next;
041
042    protected Throwable cause;
043
044    // Constructors.
045    // -----------------------------------------------------------------------
046
047    public ListenerException(Throwable cause)
048    {
049        super();
050        this.cause = cause;
051    }
052
053    public ListenerException(Throwable cause, String msg)
054    {
055        super(msg);
056        this.cause = cause;
057    }
058
059    public ListenerException(String msg)
060    {
061        super(msg);
062    }
063
064    public ListenerException()
065    {
066        super();
067    }
068
069    // Instance methods.
070    // -----------------------------------------------------------------------
071
072    /**
073     * Returns the next exception in this chain, or <code>null</code> if
074     * there are no other exceptions.
075     *
076     * @return The next exception.
077     */
078    public ListenerException getNext()
079    {
080        return next;
081    }
082
083    /**
084     * Sets the next exception in this chain.
085     *
086     * @param next The next exception.
087     */
088    public void setNext(ListenerException next)
089    {
090        this.next = next;
091    }
092
093    /**
094     * Gets the cause of this exception, or <code>null</code> if the
095     * cause is unknown.
096     *
097     * @return The cause.
098     */
099    public Throwable getCause()
100    {
101        return cause;
102    }
103
104    /**
105     * Sets the cause of this exception.
106     *
107     * @param cause The cause of this exception.
108     */
109    public synchronized Throwable initCause(Throwable cause)
110    {
111        Throwable old = this.cause;
112        this.cause = cause;
113        return old;
114    }
115}

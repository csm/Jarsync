047package org.metastatic.rsync;
048
049/**
050 * <p>An implementation of Ron Rivest's MD4 message digest algorithm.
051 * MD4 was the precursor to the stronger MD5
052 * algorithm, and while not considered cryptograpically secure itself,
053 * MD4 is in use in various applications. It is slightly faster than
054 * MD5.</p>
055 * <p/>
056 * <p>This implementation is provided for compatibility with the "MD4"
057 * implementation in the C version of rsync -- which differs in that the
058 * input is NOT padded when the input is divisible by 64, and that only
059 * the lower 32 bits of the length is used in the padding (the real MD4
060 * always pads the input and uses the entire 8 bytes in the padding).
061 * <p/>
062 * <p><strong>DO NOT USE THIS IMPLEMENTATION IN NEW
063 * PROGRAMS.</strong></p>
064 * <p/>
065 * <p>References:</p>
066 * <p/>
067 * <ol>
068 * <li>The <a href="http://www.ietf.org/rfc/rfc1320.txt">MD4</a> Message-
069 * Digest Algorithm.<br>
070 * R. Rivest.</li>
071 * </ol>
072 *
073 * @version $Revision$
074 */
075public final class BrokenMD4 extends MD4 implements Cloneable
076{
077
078    // Constants and variables.
079    // -----------------------------------------------------------------
080
081    // Constructors.
082    // -----------------------------------------------------------------
083
084    /**
085     * Trivial zero-argument constructor.
086     */
087    public BrokenMD4()
088    {
089        super();
090    }
091
092    /**
093     * Private constructor for cloning.
094     */
095    private BrokenMD4(BrokenMD4 that)
096    {
097        this();
098
099        this.a = that.a;
100        this.b = that.b;
101        this.c = that.c;
102        this.d = that.d;
103        this.count = that.count;
104        System.arraycopy(that.buffer, 0, this.buffer, 0, BLOCK_LENGTH);
105    }
106
107    // java.lang.Cloneable interface implementation --------------------
108
109    public Object clone()
110    {
111        return new BrokenMD4(this);
112    }
113
114    // SPI instance methods.
115    // -----------------------------------------------------------------
116
117    /**
118     * Pack the four chaining variables into a byte array.
119     */
120    protected byte[] engineDigest()
121    {
122        if (count % BLOCK_LENGTH != 0)
123        {
124            byte[] tail = padBuffer();
125            engineUpdate(tail, 0, tail.length);
126        }
127        byte[] digest = {
128                (byte) a, (byte) (a >>> 8), (byte) (a >>> 16), (byte) (a >>> 24),
129                (byte) b, (byte) (b >>> 8), (byte) (b >>> 16), (byte) (b >>> 24),
130                (byte) c, (byte) (c >>> 8), (byte) (c >>> 16), (byte) (c >>> 24),
131                (byte) d, (byte) (d >>> 8), (byte) (d >>> 16), (byte) (d >>> 24)
132        };
133
134        engineReset();
135
136        return digest;
137    }
138
139    /**
140     * Pad the buffer by appending the byte 0x80, then as many zero bytes
141     * to fill the buffer 8 bytes shy of being a multiple of 64 bytes, then
142     * append the length of the buffer, in bits, before padding.
143     */
144    protected byte[] padBuffer()
145    {
146        int n = (int) (count % BLOCK_LENGTH);
147        int padding = (n < 56) ? (56 - n) : (120 - n);
148        byte[] pad = new byte[padding + 8];
149
150        pad[0] = (byte) 0x80;
151        int bits = (int) (count << 3);
152        pad[padding++] = (byte) bits;
153        pad[padding++] = (byte) (bits >>> 8);
154        pad[padding++] = (byte) (bits >>> 16);
155        pad[padding++] = (byte) (bits >>> 24);
156        //pad[padding++] = (byte) (bits >>> 32);
157        //pad[padding++] = (byte) (bits >>> 40);
158        //pad[padding++] = (byte) (bits >>> 48);
159        //pad[padding  ] = (byte) (bits >>> 56);
160
161        return pad;
162    }
163}

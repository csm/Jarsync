044package org.metastatic.rsync;
045
046/**
047 * Signals that a parameter file being parsed is incorrect. This is not
048 * usually a fatal error, and parsing may continue, but the program
049 * catching this error should complain about this.
050 *
051 * @version $Revision$
052 */
053public class ParameterException extends java.io.IOException
054{
055
056    // Trivial subclass.
057    public ParameterException()
058    {
059        super();
060    }
061
062    public ParameterException(String msg)
063    {
064        super(msg);
065    }
066}

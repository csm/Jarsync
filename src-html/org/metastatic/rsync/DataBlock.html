044package org.metastatic.rsync;
045
046/**
047 * This is the {@link Delta} in the rsync algorithm that introduces new
048 * data. It is an array of bytes and an offset, such that the updated
049 * file should contain this block at the given offset.
050 *
051 * @version $Revision$
052 */
053public class DataBlock implements Delta, java.io.Serializable
054{
055
056    // Constants and variables.
057    // -----------------------------------------------------------------
058
059    private static final long serialVersionUID = -3132452687703522201L;
060
061    /**
062     * The block of data to insert.
063     *
064     * @since 1.1
065     */
066    protected final byte[] data;
067
068    /**
069     * The offset in the file to start this block.
070     *
071     * @since 1.1
072     */
073    protected final long offset;
074
075    // Constructors.
076    // -----------------------------------------------------------------
077
078    /**
079     * Create a new instance of a DataBlock with a given offset and
080     * block of bytes.
081     *
082     * @param offset The offset where this data should go.
083     * @param data   The data itself.
084     * @since 1.1
085     */
086    public DataBlock(long offset, byte[] data)
087    {
088        this.offset = offset;
089        this.data = (byte[]) data.clone();
090    }
091
092    /**
093     * Create a new instance of a DataBlock with a given offset and a
094     * portion of a byte array.
095     *
096     * @param offset The write offset of this data block.
097     * @param data   The data itself.
098     * @param off    The offset in the array to begin copying.
099     * @param len    The number of bytes to copy.
100     */
101    public DataBlock(long offset, byte[] data, int off, int len)
102    {
103        this.offset = offset;
104        if (data.length == len && off == 0)
105        {
106            this.data = (byte[]) data.clone();
107        } else
108        {
109            this.data = new byte[len];
110            System.arraycopy(data, 0, this.data, off, len);
111        }
112    }
113
114    // Instance methods.
115    // -----------------------------------------------------------------
116
117    // Delta interface implementation.
118
119    public long getWriteOffset()
120    {
121        return offset;
122    }
123
124    public int getBlockLength()
125    {
126        return data.length;
127    }
128
129    // Property accessor methods. --------------------------------------
130
131    /**
132     * Get the offset at which this block should begin.
133     *
134     * @return The offset at which this block should begin.
135     * @since 1.1
136     */
137    public long getOffset()
138    {
139        return offset;
140    }
141
142    /**
143     * Return the array of bytes that is the data block.
144     *
145     * @return The block itself.
146     * @since 1.1
147     */
148    public byte[] getData()
149    {
150        return data;
151    }
152
153    // Instance methods overriding java.lang.Object. -------------------
154
155    /**
156     * Return a printable string that represents this object.
157     *
158     * @return A string representation of this block.
159     * @since 1.1
160     */
161    public String toString()
162    {
163        String str = "[ off=" + offset + " len=" + data.length + " data=";
164        int len = Math.min(data.length, 256);
165        str += Util.toHexString(data, 0, len);
166        if (len != data.length) str += "...";
167        return str + " ]";
168    }
169
170    /**
171     * Return the hash code for this data block.
172     *
173     * @return The hash code.
174     * @since 1.1
175     */
176    public int hashCode()
177    {
178        int b = 0;
179        // For fun.
180        for (int i = 0; i < data.length; i++)
181            b ^= data[i] << ((i * 8) % 32);
182        return b + (int) offset;
183    }
184
185    /**
186     * Test if another object equals this one.
187     *
188     * @return <tt>true</tt> If <tt>o</tt> is an instance of DataBlock and
189     * if both the offsets and the byte arrays of both are equal.
190     * @throws java.lang.ClassCastException   If <tt>o</tt> is not an
191     *                                        instance of this class.
192     * @throws java.lang.NullPointerException If <tt>o</tt> is null.
193     */
194    public boolean equals(Object o)
195    {
196        return offset == ((DataBlock) o).offset &&
197                java.util.Arrays.equals(data, ((DataBlock) o).data);
198    }
199}

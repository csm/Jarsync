023package org.metastatic.rsync;
024
025/**
026 * This is the {@link Delta} in the rsync algorithm that introduces new
027 * data. It is an array of bytes and an offset, such that the updated
028 * file should contain this block at the given offset.
029 *
030 * @version $Revision$
031 */
032public class DataBlock implements Delta, java.io.Serializable
033{
034
035    // Constants and variables.
036    // -----------------------------------------------------------------
037
038    private static final long serialVersionUID = -3132452687703522201L;
039
040    /**
041     * The block of data to insert.
042     *
043     * @since 1.1
044     */
045    protected final byte[] data;
046
047    /**
048     * The offset in the file to start this block.
049     *
050     * @since 1.1
051     */
052    protected final long offset;
053
054    // Constructors.
055    // -----------------------------------------------------------------
056
057    /**
058     * Create a new instance of a DataBlock with a given offset and
059     * block of bytes.
060     *
061     * @param offset The offset where this data should go.
062     * @param data   The data itself.
063     * @since 1.1
064     */
065    public DataBlock(long offset, byte[] data)
066    {
067        this.offset = offset;
068        this.data = (byte[]) data.clone();
069    }
070
071    /**
072     * Create a new instance of a DataBlock with a given offset and a
073     * portion of a byte array.
074     *
075     * @param offset The write offset of this data block.
076     * @param data   The data itself.
077     * @param off    The offset in the array to begin copying.
078     * @param len    The number of bytes to copy.
079     */
080    public DataBlock(long offset, byte[] data, int off, int len)
081    {
082        this.offset = offset;
083        if (data.length == len && off == 0)
084        {
085            this.data = (byte[]) data.clone();
086        } else
087        {
088            this.data = new byte[len];
089            System.arraycopy(data, 0, this.data, off, len);
090        }
091    }
092
093    // Instance methods.
094    // -----------------------------------------------------------------
095
096    // Delta interface implementation.
097
098    public long getWriteOffset()
099    {
100        return offset;
101    }
102
103    public int getBlockLength()
104    {
105        return data.length;
106    }
107
108    // Property accessor methods. --------------------------------------
109
110    /**
111     * Get the offset at which this block should begin.
112     *
113     * @return The offset at which this block should begin.
114     * @since 1.1
115     */
116    public long getOffset()
117    {
118        return offset;
119    }
120
121    /**
122     * Return the array of bytes that is the data block.
123     *
124     * @return The block itself.
125     * @since 1.1
126     */
127    public byte[] getData()
128    {
129        return data;
130    }
131
132    // Instance methods overriding java.lang.Object. -------------------
133
134    /**
135     * Return a printable string that represents this object.
136     *
137     * @return A string representation of this block.
138     * @since 1.1
139     */
140    public String toString()
141    {
142        String str = "DataBlock(off=" + offset + " len=" + data.length + " data=";
143        int len = Math.min(data.length, 256);
144        str += Util.toHexString(data, 0, len);
145        if (len != data.length) str += "...";
146        return str + ")";
147    }
148
149    /**
150     * Return the hash code for this data block.
151     *
152     * @return The hash code.
153     * @since 1.1
154     */
155    public int hashCode()
156    {
157        int b = 0;
158        // For fun.
159        for (int i = 0; i < data.length; i++)
160            b ^= data[i] << ((i * 8) % 32);
161        return b + (int) offset;
162    }
163
164    /**
165     * Test if another object equals this one.
166     *
167     * @return <tt>true</tt> If <tt>o</tt> is an instance of DataBlock and
168     * if both the offsets and the byte arrays of both are equal.
169     * @throws java.lang.ClassCastException   If <tt>o</tt> is not an
170     *                                        instance of this class.
171     * @throws java.lang.NullPointerException If <tt>o</tt> is null.
172     */
173    public boolean equals(Object o)
174    {
175        return offset == ((DataBlock) o).offset &&
176                java.util.Arrays.equals(data, ((DataBlock) o).data);
177    }
178}

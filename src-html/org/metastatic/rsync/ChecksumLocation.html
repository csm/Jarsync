023package org.metastatic.rsync;
024
025import com.google.common.base.Preconditions;
026
027/**
028 * A {@link org.metastatic.rsync.ChecksumPair} and information about the location
029 * that pair was generated over.
030 */
031public class ChecksumLocation
032{
033    private final ChecksumPair checksumPair;
034
035    private final long offset;
036    private final int length;
037    private final int seq;
038
039    /**
040     * Create a checksum location.
041     *
042     * @param checksumPair The checksum pair.
043     * @param offset The offset of the data the checksums were computed over.
044     * @param length The length of the data.
045     * @param seq A sequence number for this location.
046     */
047    public ChecksumLocation(ChecksumPair checksumPair, long offset, int length, int seq)
048    {
049        this.checksumPair = Preconditions.checkNotNull(checksumPair);
050        this.offset = offset;
051        this.length = length;
052        this.seq = seq;
053    }
054
055    /**
056     * Create a checksum location. The sequence will default to zero.
057     *
058     * @param checksumPair The checksum pair.
059     * @param offset The offset of the data the checksums were computed over.
060     * @param length The length of the data.
061     */
062    public ChecksumLocation(ChecksumPair checksumPair, long offset, int length)
063    {
064        this(checksumPair, offset, length, 0);
065    }
066
067    /**
068     * Create a checksum location. The sequence and length will default to zero.
069     *
070     * @param checksumPair The checksum pair.
071     * @param offset The offset of the data the checksums were computed over.
072     */
073    public ChecksumLocation(ChecksumPair checksumPair, long offset)
074    {
075        this(checksumPair, offset, 0, 0);
076    }
077
078    /**
079     * Get the checksum pair.
080     *
081     * @return The checksum pair.
082     */
083    public ChecksumPair getChecksumPair()
084    {
085        return checksumPair;
086    }
087
088    /**
089     * Get the data offset.
090     *
091     * @return The offset.
092     */
093    public long getOffset()
094    {
095        return offset;
096    }
097
098    /**
099     * Get the data length.
100     *
101     * @return The length.
102     */
103    public int getLength()
104    {
105        return length;
106    }
107
108    /**
109     * Get the sequence number.
110     *
111     * @return The sequence number.
112     */
113    public int getSeq()
114    {
115        return seq;
116    }
117
118    @Override
119    public boolean equals(Object o)
120    {
121        if (this == o) return true;
122        if (o == null || getClass() != o.getClass()) return false;
123
124        ChecksumLocation that = (ChecksumLocation) o;
125
126        if (length != that.length) return false;
127        if (offset != that.offset) return false;
128        if (seq != that.seq) return false;
129        if (checksumPair != null ? !checksumPair.equals(that.checksumPair) : that.checksumPair != null) return false;
130
131        return true;
132    }
133
134    @Override
135    public int hashCode()
136    {
137        int result = checksumPair != null ? checksumPair.hashCode() : 0;
138        result = 31 * result + (int) (offset ^ (offset >>> 32));
139        result = 31 * result + length;
140        result = 31 * result + seq;
141        return result;
142    }
143
144    @Override
145    public String toString()
146    {
147        final StringBuilder sb = new StringBuilder("ChecksumLocation(");
148        sb.append("checksumPair=").append(checksumPair);
149        sb.append(", offset=").append(offset);
150        sb.append(", length=").append(length);
151        sb.append(", seq=").append(seq);
152        sb.append(')');
153        return sb.toString();
154    }
155}

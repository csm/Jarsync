024package org.metastatic.rsync;
025
026/**
027 * A simple 32-bit "rolling" checksum. This checksum algorithm is based
028 * upon the algorithm outlined in the paper "The rsync algorithm" by
029 * Andrew Tridgell and Paul Mackerras. The algorithm works in such a way
030 * that if one knows the sum of a block
031 * <em>X<sub>k</sub>...X<sub>l</sub></em>, then it is a simple matter to
032 * compute the sum for <em>X<sub>k+1</sub>...X<sub>l+1</sub></em>.
033 *
034 * @author Casey Marshall
035 * @version $Revision$
036 */
037public class Checksum32 implements RollingChecksum, Cloneable, java.io.Serializable
038{
039
040    // Constants and variables.
041    // -----------------------------------------------------------------
042
043    protected final int char_offset;
044
045    /**
046     * The first half of the checksum.
047     *
048     * @since 1.1
049     */
050    protected int a;
051
052    /**
053     * The second half of the checksum.
054     *
055     * @since 1.1
056     */
057    protected int b;
058
059    /**
060     * The place from whence the current checksum has been computed.
061     *
062     * @since 1.1
063     */
064    protected int k;
065
066    /**
067     * The place to where the current checksum has been computed.
068     *
069     * @since 1.1
070     */
071    protected int l;
072
073    /**
074     * The block from which the checksum is computed.
075     *
076     * @since 1.1
077     */
078    protected byte[] block;
079
080    /**
081     * The index in {@link #new_block} where the newest byte has
082     * been stored.
083     *
084     * @since 1.1
085     */
086    protected int new_index;
087
088    /**
089     * The block that is recieving new input.
090     *
091     * @since 1.1
092     */
093    protected byte[] new_block;
094
095    // Constructors.
096    // -----------------------------------------------------------------
097
098    /**
099     * Creates a new rolling checksum. The <i>char_offset</i> argument
100     * affects the output of this checksum; rsync uses a char offset of
101     * 0, librsync 31.
102     */
103    public Checksum32(int char_offset)
104    {
105        this.char_offset = char_offset;
106        a = b = 0;
107        k = 0;
108    }
109
110    public Checksum32()
111    {
112        this(0);
113    }
114
115    private Checksum32(Checksum32 that)
116    {
117        this.char_offset = that.char_offset;
118        this.a = that.a;
119        this.b = that.b;
120        this.l = that.l;
121        this.k = that.k;
122        this.block = (that.block != null) ? (byte[]) ((byte[]) that.block).clone() : null;
123        this.new_index = that.new_index;
124        this.new_block = (that.new_block != null)
125                ? (byte[]) ((byte[]) that.new_block).clone() : null;
126    }
127
128    // Public instance methods.
129    // -----------------------------------------------------------------
130
131    /**
132     * Return the value of the currently computed checksum.
133     *
134     * @return The currently computed checksum.
135     * @since 1.1
136     */
137    public int getValue()
138    {
139        return (a & 0xffff) | (b << 16);
140    }
141
142    /**
143     * Reset the checksum.
144     *
145     * @since 1.1
146     */
147    public void reset()
148    {
149        k = 0;
150        a = b = 0;
151        l = 0;
152    }
153
154    /**
155     * "Roll" the checksum. This method takes a single byte as byte
156     * <em>X<sub>l+1</sub></em>, and recomputes the checksum for
157     * <em>X<sub>k+1</sub>...X<sub>l+1</sub></em>. This is the
158     * preferred method for updating the checksum.
159     *
160     * @param bt The next byte.
161     * @since 1.1
162     */
163    public void roll(byte bt)
164    {
165        a -= block[k] + char_offset;
166        b -= l * (block[k] + char_offset);
167        a += bt + char_offset;
168        b += a;
169        block[k] = bt;
170        k++;
171        if (k == l) k = 0;
172    }
173
174    /**
175     * Update the checksum by trimming off a byte only, not adding
176     * anything.
177     */
178    public void trim()
179    {
180        a -= block[k % block.length] + char_offset;
181        b -= l * (block[k % block.length] + char_offset);
182        k++;
183        l--;
184    }
185
186    /**
187     * Update the checksum with an entirely different block, and
188     * potentially a different block length.
189     *
190     * @param buf The byte array that holds the new block.
191     * @param off From whence to begin reading.
192     * @param len The length of the block to read.
193     * @since 1.1
194     */
195    public void check(byte[] buf, int off, int len)
196    {
197        block = new byte[len];
198        System.arraycopy(buf, off, block, 0, len);
199        reset();
200        l = block.length;
201        int i;
202
203        for (i = 0; i < block.length - 4; i += 4)
204        {
205            b += 4 * (a + block[i]) + 3 * block[i + 1] +
206                    2 * block[i + 2] + block[i + 3] + 10 * char_offset;
207            a += block[i] + block[i + 1] + block[i + 2]
208                    + block[i + 3] + 4 * char_offset;
209        }
210        for (; i < block.length; i++)
211        {
212            a += block[i] + char_offset;
213            b += a;
214        }
215    }
216
217    public Object clone()
218    {
219        try
220        {
221            return super.clone();
222        } catch (CloneNotSupportedException cnse)
223        {
224            throw new Error();
225        }
226    }
227
228    public boolean equals(Object o)
229    {
230        return ((Checksum32) o).a == a && ((Checksum32) o).b == b;
231    }
232}

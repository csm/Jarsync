043package org.metastatic.rsync;
044
045/**
046 * A simple 32-bit "rolling" checksum. This checksum algorithm is based
047 * upon the algorithm outlined in the paper "The rsync algorithm" by
048 * Andrew Tridgell and Paul Mackerras. The algorithm works in such a way
049 * that if one knows the sum of a block
050 * <em>X<sub>k</sub>...X<sub>l</sub></em>, then it is a simple matter to
051 * compute the sum for <em>X<sub>k+1</sub>...X<sub>l+1</sub></em>.
052 *
053 * @author Casey Marshall
054 * @version $Revision$
055 */
056public class Checksum32 implements RollingChecksum, Cloneable, java.io.Serializable
057{
058
059    // Constants and variables.
060    // -----------------------------------------------------------------
061
062    protected final int char_offset;
063
064    /**
065     * The first half of the checksum.
066     *
067     * @since 1.1
068     */
069    protected int a;
070
071    /**
072     * The second half of the checksum.
073     *
074     * @since 1.1
075     */
076    protected int b;
077
078    /**
079     * The place from whence the current checksum has been computed.
080     *
081     * @since 1.1
082     */
083    protected int k;
084
085    /**
086     * The place to where the current checksum has been computed.
087     *
088     * @since 1.1
089     */
090    protected int l;
091
092    /**
093     * The block from which the checksum is computed.
094     *
095     * @since 1.1
096     */
097    protected byte[] block;
098
099    /**
100     * The index in {@link #new_block} where the newest byte has
101     * been stored.
102     *
103     * @since 1.1
104     */
105    protected int new_index;
106
107    /**
108     * The block that is recieving new input.
109     *
110     * @since 1.1
111     */
112    protected byte[] new_block;
113
114    // Constructors.
115    // -----------------------------------------------------------------
116
117    /**
118     * Creates a new rolling checksum. The <i>char_offset</i> argument
119     * affects the output of this checksum; rsync uses a char offset of
120     * 0, librsync 31.
121     */
122    public Checksum32(int char_offset)
123    {
124        this.char_offset = char_offset;
125        a = b = 0;
126        k = 0;
127    }
128
129    public Checksum32()
130    {
131        this(0);
132    }
133
134    private Checksum32(Checksum32 that)
135    {
136        this.char_offset = that.char_offset;
137        this.a = that.a;
138        this.b = that.b;
139        this.l = that.l;
140        this.k = that.k;
141        this.block = (that.block != null) ? (byte[]) ((byte[]) that.block).clone() : null;
142        this.new_index = that.new_index;
143        this.new_block = (that.new_block != null)
144                ? (byte[]) ((byte[]) that.new_block).clone() : null;
145    }
146
147    // Public instance methods.
148    // -----------------------------------------------------------------
149
150    /**
151     * Return the value of the currently computed checksum.
152     *
153     * @return The currently computed checksum.
154     * @since 1.1
155     */
156    public int getValue()
157    {
158        return (a & 0xffff) | (b << 16);
159    }
160
161    /**
162     * Reset the checksum.
163     *
164     * @since 1.1
165     */
166    public void reset()
167    {
168        k = 0;
169        a = b = 0;
170        l = 0;
171    }
172
173    /**
174     * "Roll" the checksum. This method takes a single byte as byte
175     * <em>X<sub>l+1</sub></em>, and recomputes the checksum for
176     * <em>X<sub>k+1</sub>...X<sub>l+1</sub></em>. This is the
177     * preferred method for updating the checksum.
178     *
179     * @param bt The next byte.
180     * @since 1.1
181     */
182    public void roll(byte bt)
183    {
184        a -= block[k] + char_offset;
185        b -= l * (block[k] + char_offset);
186        a += bt + char_offset;
187        b += a;
188        block[k] = bt;
189        k++;
190        if (k == l) k = 0;
191    }
192
193    /**
194     * Update the checksum by trimming off a byte only, not adding
195     * anything.
196     */
197    public void trim()
198    {
199        a -= block[k % block.length] + char_offset;
200        b -= l * (block[k % block.length] + char_offset);
201        k++;
202        l--;
203    }
204
205    /**
206     * Update the checksum with an entirely different block, and
207     * potentially a different block length.
208     *
209     * @param buf The byte array that holds the new block.
210     * @param off From whence to begin reading.
211     * @param len The length of the block to read.
212     * @since 1.1
213     */
214    public void check(byte[] buf, int off, int len)
215    {
216        block = new byte[len];
217        System.arraycopy(buf, off, block, 0, len);
218        reset();
219        l = block.length;
220        int i;
221
222        for (i = 0; i < block.length - 4; i += 4)
223        {
224            b += 4 * (a + block[i]) + 3 * block[i + 1] +
225                    2 * block[i + 2] + block[i + 3] + 10 * char_offset;
226            a += block[i] + block[i + 1] + block[i + 2]
227                    + block[i + 3] + 4 * char_offset;
228        }
229        for (; i < block.length; i++)
230        {
231            a += block[i] + char_offset;
232            b += a;
233        }
234    }
235
236    public Object clone()
237    {
238        try
239        {
240            return super.clone();
241        } catch (CloneNotSupportedException cnse)
242        {
243            throw new Error();
244        }
245    }
246
247    public boolean equals(Object o)
248    {
249        return ((Checksum32) o).a == a && ((Checksum32) o).b == b;
250    }
251}

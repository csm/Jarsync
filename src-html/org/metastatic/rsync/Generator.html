044package org.metastatic.rsync;
045
046import java.io.File;
047import java.io.FileInputStream;
048import java.io.IOException;
049import java.io.InputStream;
050
051import java.util.ArrayList;
052import java.util.List;
053
054/**
055 * A checksum generator. This class can take a byte array or an input stream,
056 * and will generate a list of {@link org.metastatic.rsync.ChecksumPair} objects
057 * for that data.
058 */
059public class Generator
060{
061
062    // Constants and variables.
063    // ------------------------------------------------------------------------
064
065    /**
066     * Our configuration. Contains such things as our rolling checksum
067     * and message digest.
068     */
069    protected final Configuration config;
070
071    // Constructors.
072    // ------------------------------------------------------------------------
073
074    public Generator(Configuration config)
075    {
076        this.config = config;
077    }
078
079    // Instance methods.
080    // ------------------------------------------------------------------------
081
082    /**
083     * Generate checksums over an entire byte array, with a base offset
084     * of 0.
085     *
086     * @param buf The byte buffer to checksum.
087     * @return A {@link java.util.List} of {@link ChecksumPair}s
088     * generated from the array.
089     * @see #generateSums(byte[], int, int, long)
090     */
091    public List<ChecksumPair> generateSums(byte[] buf)
092    {
093        return generateSums(buf, 0, buf.length, 0);
094    }
095
096    /**
097     * Generate checksums over a portion of a byte array, with a base
098     * offset of 0.
099     *
100     * @param buf The byte array to checksum.
101     * @param off The offset in <code>buf</code> to begin.
102     * @param len The number of bytes to checksum.
103     * @return A {@link java.util.List} of {@link ChecksumPair}s
104     * generated from the array.
105     * @see #generateSums(byte[], int, int, long)
106     */
107    public List<ChecksumPair> generateSums(byte[] buf, int off, int len)
108    {
109        return generateSums(buf, off, len, 0);
110    }
111
112    /**
113     * Generate checksums over an entire byte array, with a specified
114     * base offset. This <code>baseOffset</code> is added to the offset
115     * stored in each {@link ChecksumPair}.
116     *
117     * @param buf        The byte array to checksum.
118     * @param baseOffset The offset from whence this byte array came.
119     * @return A {@link java.util.List} of {@link ChecksumPair}s
120     * generated from the array.
121     * @see #generateSums(byte[], int, int, long)
122     */
123    public List<ChecksumPair> generateSums(byte[] buf, long baseOffset)
124    {
125        return generateSums(buf, 0, buf.length, baseOffset);
126    }
127
128    /**
129     * Generate checksums over a portion of abyte array, with a specified
130     * base offset. This <code>baseOffset</code> is added to the offset
131     * stored in each {@link ChecksumPair}.
132     *
133     * @param buf        The byte array to checksum.
134     * @param off        From whence in <code>buf</code> to start.
135     * @param len        The number of bytes to check in
136     *                   <code>buf</code>.
137     * @param baseOffset The offset from whence this byte array came.
138     * @return A {@link java.util.List} of {@link ChecksumPair}s
139     * generated from the array.
140     */
141    public List<ChecksumPair> generateSums(byte[] buf, int off, int len, long baseOffset)
142    {
143        int count = (len + (config.blockLength - 1)) / config.blockLength;
144        int remainder = len % config.blockLength;
145        int offset = off;
146        List<ChecksumPair> sums = new ArrayList<ChecksumPair>(count);
147
148        for (int i = 0; i < count; i++)
149        {
150            int n = Math.min(len, config.blockLength);
151            ChecksumPair pair = generateSum(buf, offset, n, offset + baseOffset);
152            pair.seq = i;
153
154            sums.add(pair);
155            len -= n;
156            offset += n;
157        }
158
159        return sums;
160    }
161
162    /**
163     * Generate checksums for an entire file.
164     *
165     * @param f The {@link java.io.File} to checksum.
166     * @return A {@link java.util.List} of {@link ChecksumPair}s
167     * generated from the file.
168     * @throws java.io.IOException if <code>f</code> cannot be read from.
169     */
170    public List<ChecksumPair> generateSums(File f) throws IOException
171    {
172        long len = f.length();
173        int count = (int) ((len + (config.blockLength + 1)) / config.blockLength);
174        long offset = 0;
175        FileInputStream fin = new FileInputStream(f);
176        List<ChecksumPair> sums = new ArrayList<ChecksumPair>(count);
177        int n = (int) Math.min(len, config.blockLength);
178        byte[] buf = new byte[n];
179
180        for (int i = 0; i < count; i++)
181        {
182            int l = fin.read(buf, 0, n);
183            if (l == -1) break;
184            ChecksumPair pair = generateSum(buf, 0, Math.min(l, n), offset);
185            pair.seq = i;
186
187            sums.add(pair);
188            len -= n;
189            offset += n;
190            n = (int) Math.min(len, config.blockLength);
191        }
192
193        fin.close();
194        return sums;
195    }
196
197    /**
198     * Generate checksums for an InputStream.
199     *
200     * @param in The {@link java.io.InputStream} to checksum.
201     * @return A {@link java.util.List} of {@link ChecksumPair}s
202     * generated from the bytes read.
203     * @throws java.io.IOException if reading fails.
204     */
205    public List<ChecksumPair> generateSums(InputStream in) throws IOException
206    {
207        List<ChecksumPair> sums = null;
208        byte[] buf = new byte[config.blockLength * config.blockLength];
209        long offset = 0;
210        int len = 0;
211
212        while ((len = in.read(buf)) != -1)
213        {
214            if (sums == null)
215            {
216                sums = generateSums(buf, 0, len, offset);
217            } else
218            {
219                sums.addAll(generateSums(buf, 0, len, offset));
220            }
221            offset += len;
222        }
223
224        return sums;
225    }
226
227    /**
228     * Generate a sum pair for an entire byte array.
229     *
230     * @param buf        The byte array to checksum.
231     * @param fileOffset The offset in the original file from whence
232     *                   this block came.
233     * @return A {@link ChecksumPair} for this byte array.
234     */
235    public ChecksumPair generateSum(byte[] buf, long fileOffset)
236    {
237        return generateSum(buf, 0, buf.length, fileOffset);
238    }
239
240    /**
241     * Generate a sum pair for a portion of a byte array.
242     *
243     * @param buf        The byte array to checksum.
244     * @param off        Where in <code>buf</code> to start.
245     * @param len        How many bytes to checksum.
246     * @param fileOffset The original offset of this byte array.
247     * @return A {@link ChecksumPair} for this byte array.
248     */
249    public ChecksumPair
250    generateSum(byte[] buf, int off, int len, long fileOffset)
251    {
252        ChecksumPair p = new ChecksumPair();
253        config.weakSum.check(buf, off, len);
254        config.strongSum.update(buf, off, len);
255        if (config.checksumSeed != null)
256        {
257            config.strongSum.update(config.checksumSeed, 0,
258                    config.checksumSeed.length);
259        }
260        p.weak = config.weakSum.getValue();
261        p.strong = new byte[config.strongSumLength];
262        System.arraycopy(config.strongSum.digest(), 0, p.strong, 0,
263                p.strong.length);
264        p.offset = fileOffset;
265        p.length = len;
266        return p;
267    }

024package org.metastatic.rsync;
025
026import java.io.File;
027import java.io.FileInputStream;
028import java.io.IOException;
029import java.io.InputStream;
030
031import java.util.ArrayList;
032import java.util.List;
033
034/**
035 * A checksum generator. This class can take a byte array or an input stream,
036 * and will generate a list of {@link org.metastatic.rsync.ChecksumPair} objects
037 * for that data.
038 */
039public class Generator
040{
041
042    // Constants and variables.
043    // ------------------------------------------------------------------------
044
045    /**
046     * Our configuration. Contains such things as our rolling checksum
047     * and message digest.
048     */
049    protected final Configuration config;
050
051    // Constructors.
052    // ------------------------------------------------------------------------
053
054    public Generator(Configuration config)
055    {
056        this.config = config;
057    }
058
059    // Instance methods.
060    // ------------------------------------------------------------------------
061
062    /**
063     * Generate checksums over an entire byte array, with a base offset
064     * of 0.
065     *
066     * @param buf The byte buffer to checksum.
067     * @return A {@link java.util.List} of {@link ChecksumPair}s
068     * generated from the array.
069     * @see #generateSums(byte[], int, int, long)
070     */
071    public List<ChecksumLocation> generateSums(byte[] buf)
072    {
073        return generateSums(buf, 0, buf.length, 0);
074    }
075
076    /**
077     * Generate checksums over a portion of a byte array, with a base
078     * offset of 0.
079     *
080     * @param buf The byte array to checksum.
081     * @param off The offset in <code>buf</code> to begin.
082     * @param len The number of bytes to checksum.
083     * @return A {@link java.util.List} of {@link ChecksumPair}s
084     * generated from the array.
085     * @see #generateSums(byte[], int, int, long)
086     */
087    public List<ChecksumLocation> generateSums(byte[] buf, int off, int len)
088    {
089        return generateSums(buf, off, len, 0);
090    }
091
092    /**
093     * Generate checksums over an entire byte array, with a specified
094     * base offset. This <code>baseOffset</code> is added to the offset
095     * stored in each {@link ChecksumPair}.
096     *
097     * @param buf        The byte array to checksum.
098     * @param baseOffset The offset from whence this byte array came.
099     * @return A {@link java.util.List} of {@link ChecksumPair}s
100     * generated from the array.
101     * @see #generateSums(byte[], int, int, long)
102     */
103    public List<ChecksumLocation> generateSums(byte[] buf, long baseOffset)
104    {
105        return generateSums(buf, 0, buf.length, baseOffset);
106    }
107
108    /**
109     * Generate checksums over a portion of abyte array, with a specified
110     * base offset. This <code>baseOffset</code> is added to the offset
111     * stored in each {@link ChecksumPair}.
112     *
113     * @param buf        The byte array to checksum.
114     * @param off        From whence in <code>buf</code> to start.
115     * @param len        The number of bytes to check in
116     *                   <code>buf</code>.
117     * @param baseOffset The offset from whence this byte array came.
118     * @return A {@link java.util.List} of {@link ChecksumPair}s
119     * generated from the array.
120     */
121    public List<ChecksumLocation> generateSums(byte[] buf, int off, int len, long baseOffset)
122    {
123        int count = (len + (config.blockLength - 1)) / config.blockLength;
124        int remainder = len % config.blockLength;
125        int offset = off;
126        List<ChecksumLocation> sums = new ArrayList<ChecksumLocation>(count);
127
128        for (int i = 0; i < count; i++)
129        {
130            int n = Math.min(len, config.blockLength);
131            ChecksumLocation location = generateSum(buf, offset, n, offset + baseOffset, i);
132
133            sums.add(location);
134            len -= n;
135            offset += n;
136        }
137
138        return sums;
139    }
140
141    /**
142     * Generate checksums for an entire file.
143     *
144     * @param f The {@link java.io.File} to checksum.
145     * @return A {@link java.util.List} of {@link org.metastatic.rsync.ChecksumLocation}s
146     * generated from the file.
147     * @throws java.io.IOException if <code>f</code> cannot be read from.
148     */
149    public List<ChecksumLocation> generateSums(File f) throws IOException
150    {
151        long len = f.length();
152        int count = (int) ((len + (config.blockLength + 1)) / config.blockLength);
153        long offset = 0;
154        FileInputStream fin = new FileInputStream(f);
155        List<ChecksumLocation> sums = new ArrayList<ChecksumLocation>(count);
156        int n = (int) Math.min(len, config.blockLength);
157        byte[] buf = new byte[n];
158
159        for (int i = 0; i < count; i++)
160        {
161            int l = fin.read(buf, 0, n);
162            if (l == -1) break;
163            ChecksumLocation location = generateSum(buf, 0, Math.min(l, n), offset, i);
164
165            sums.add(location);
166            len -= n;
167            offset += n;
168            n = (int) Math.min(len, config.blockLength);
169        }
170
171        fin.close();
172        return sums;
173    }
174
175    /**
176     * Generate checksums for an InputStream.
177     *
178     * @param in The {@link java.io.InputStream} to checksum.
179     * @return A {@link java.util.List} of {@link org.metastatic.rsync.ChecksumLocation}s
180     * generated from the bytes read.
181     * @throws java.io.IOException if reading fails.
182     */
183    public List<ChecksumLocation> generateSums(InputStream in) throws IOException
184    {
185        List<ChecksumLocation> sums = null;
186        byte[] buf = new byte[config.blockLength * config.blockLength];
187        long offset = 0;
188        int len = 0;
189
190        while ((len = in.read(buf)) != -1)
191        {
192            if (sums == null)
193            {
194                sums = generateSums(buf, 0, len, offset);
195            } else
196            {
197                sums.addAll(generateSums(buf, 0, len, offset));
198            }
199            offset += len;
200        }
201
202        return sums;
203    }
204
205    /**
206     * Generate a sum pair for an entire byte array.
207     *
208     * @param buf        The byte array to checksum.
209     * @param fileOffset The offset in the original file from whence
210     *                   this block came.
211     * @return A {@link org.metastatic.rsync.ChecksumLocation} for this byte array.
212     */
213    public ChecksumLocation generateSum(byte[] buf, long fileOffset)
214    {
215        return generateSum(buf, 0, buf.length, fileOffset);
216    }
217
218    public ChecksumLocation generateSum(byte[] buf, int off, int len, long fileOffset)
219    {
220        return generateSum(buf, off, len, fileOffset, 0);
221    }
222
223    /**
224     * Generate a sum pair for a portion of a byte array.
225     *
226     * @param buf        The byte array to checksum.
227     * @param off        Where in <code>buf</code> to start.
228     * @param len        How many bytes to checksum.
229     * @param fileOffset The original offset of this byte array.
230     * @param seq        The sequence number of this sum.
231     * @return A {@link org.metastatic.rsync.ChecksumLocation} for this byte array.
232     */
233    public ChecksumLocation generateSum(byte[] buf, int off, int len, long fileOffset, int seq)
234    {
235        config.weakSum.check(buf, off, len);
236        if (config.checksumSeed != null && config.isSeedPrefix)
237            config.strongSum.update(config.checksumSeed);
238        config.strongSum.update(buf, off, len);
239        if (config.checksumSeed != null && !config.isSeedPrefix)
240            config.strongSum.update(config.checksumSeed);
241        int weak = config.weakSum.getValue();
242        byte[] strong = new byte[config.strongSumLength];
243        System.arraycopy(config.strongSum.digest(), 0, strong, 0, config.strongSumLength);
244        return new ChecksumLocation(new ChecksumPair(weak, strong), fileOffset, len, seq);
245    }

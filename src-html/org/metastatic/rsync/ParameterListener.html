023package org.metastatic.rsync;
024
025/**
026 * A listener for parsing Samba-style config files. There are two events
027 * that occur during parsing: starting named sections and setting
028 * parameters.
029 *
030 * @version $Revision$
031 */
032public interface ParameterListener extends java.util.EventListener
033{
034
035    /**
036     * Begin a named section. This method will be called on every new
037     * section definition.
038     *
039     * @param sectionName The new section's name.
040     */
041    void beginSection(String sectionName);
042
043    /**
044     * Set a parameter. This method may be called prior to the first call
045     * to {@link #beginSection}, if there are global options. If this
046     * method is called after a call to {@link
047     * #beginSection(java.lang.String)}, this parameter belongs to that
048     * section.
049     *
050     * @param name  The parameter's name.
051     * @param value The parameter's value.
052     */
053    void setParameter(String name, String value);
054}

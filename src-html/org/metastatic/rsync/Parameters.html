044package org.metastatic.rsync;
045
046import java.io.IOException;
047import java.io.FileReader;
048import java.io.LineNumberReader;
049import java.util.StringTokenizer;
050
051/**
052 * <p>Simple parser for Samba-style config files. The parameters are
053 * passed back to the caller via a simple event listener callback
054 * interface, {@link ParameterListener}.</p>
055 * <p/>
056 * <p>A sample file looks like:</p>
057 * <p/>
058 * <pre>
059 * [section one]
060 * parameter one = value string
061 * parameter two = another value
062 * [section two]
063 * new parameter = some value or t'other
064 * </pre>
065 * <p/>
066 * <p>The syntax is roughly:</p>
067 * <p/>
068 * <pre>
069 * file      ::= parameter* section* EOF
070 * section   ::= header parameter*
071 * header    ::= '[' NAME ']'
072 * parameter ::= NAME '=' VALUE EOL
073 * </pre>
074 * <p/>
075 * <p>Blank lines, and lines that begin with either '#' or ';' are
076 * ignored. long lines may be continued by preceding the end-of-line
077 * character(s) with a backslash ('\').</p>
078 *
079 * @version $Revision$
080 */
081public final class Parameters
082{
083
084    // Constants and fields.
085    // -----------------------------------------------------------------------
086
087    /**
088     * The callback object.
089     */
090    private final ParameterListener listener;
091
092    /**
093     * The current file being read.
094     */
095    private LineNumberReader in;
096
097    // Constructor.
098    // -----------------------------------------------------------------------
099
100    /**
101     * Create a new parameter file parser. The argument is a concrete
102     * imlpmentation of {@link ParameterListener} which will take the
103     * parsed arguments.
104     *
105     * @param listener The parameter listener.
106     */
107    public Parameters(ParameterListener listener)
108    {
109        this.listener = listener;
110    }
111
112    // Instance methods.
113    // -----------------------------------------------------------------------
114
115    /**
116     * Begin parsing file <i>filename</i>.
117     *
118     * @param filename The name of the file to parse.
119     * @throws IOException If an I/O error occurs.
120     */
121    public void begin(String filename) throws IOException
122    {
123        in = new LineNumberReader(new FileReader(filename));
124    }
125
126    /**
127     * Parse, or continue parsing the file if a parsing error occured
128     * in a previous call to this method. A call to {@link
129     * #begin(java.lang.String)} must have succeeded before this method
130     * is called.
131     *
132     * @throws ParameterException If a parsing error occurs. Parsing can
133     *                            continue if this exception is thrown by calling this
134     *                            method again.
135     * @throws IOException        If an I/O error occurs.
136     */
137    public void parse() throws IOException
138    {
139        if (in == null)
140        {
141            throw new IOException("nothing to parse");
142        }
143        String line;
144
145        while ((line = in.readLine()) != null)
146        {
147            if (isIgnorable(line)) continue;
148
149            // Concatenate continuation lines.
150            while (line.endsWith("\\"))
151            {
152                String line2 = in.readLine();
153                if (line2 == null) break;
154                if (isIgnorable(line2)) continue;
155                line = line.substring(0, line.length() - 1);
156
157                // Make sure we aren't fooled by '\' followed by a space.
158                if (line2.endsWith("\\"))
159                {
160                    line += line2.trim();
161                } else if (line2.trim().endsWith("\\"))
162                {
163                    line2 = line2.trim() + " ";
164                    line += line2;
165                } else
166                {
167                    line += line2.trim();
168                }
169            }
170
171            line = line.trim();
172            int i;
173            if (line.startsWith("[") && line.endsWith("]"))
174            {
175                listener.beginSection(crush(line.substring(1, line.length() - 1)));
176            } else if ((i = line.indexOf('=')) > 1)
177            {
178                listener.setParameter(crush(line.substring(0, i).trim()),
179                        crush(line.substring(i + 1).trim()));
180            } else
181            {
182                throw new ParameterException("malformed line at "
183                        + in.getLineNumber());
184            }
185        }
186
187        in.close();
188        in = null;
189    }
190
191    // Own methods.
192    // -----------------------------------------------------------------------
193
194    /**
195     * Test if a line is a comment or empty.
196     *
197     * @param s The string to test.
198     * @return true if this string is a comment or empty.
199     */
200    private static boolean isIgnorable(String s)
201    {
202        s = s.trim();
203        if (s.length() == 0) return true;
204        if (s.charAt(0) == '#' || s.charAt(0) == ';') return true;
205        return false;
206    }
207
208    /**
209     * Crush multiple spaces (or other whitespace) into a single space.
210     *
211     * @param s The string to crush.
212     * @return The crushed string.
213     */
214    private static String crush(String s)
215    {
216        StringTokenizer tok = new StringTokenizer(s);
217        if (tok.countTokens() <= 1) return s;
218        String result = tok.nextToken();
219        while (tok.hasMoreTokens())
220            result += " " + tok.nextToken();
221        return result;
222    }
223}

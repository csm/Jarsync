023package org.metastatic.rsync;
024
025import java.io.IOException;
026import java.io.FileReader;
027import java.io.LineNumberReader;
028import java.util.StringTokenizer;
029
030/**
031 * <p>Simple parser for Samba-style config files. The parameters are
032 * passed back to the caller via a simple event listener callback
033 * interface, {@link ParameterListener}.</p>
034 * <p/>
035 * <p>A sample file looks like:</p>
036 * <p/>
037 * <pre>
038 * [section one]
039 * parameter one = value string
040 * parameter two = another value
041 * [section two]
042 * new parameter = some value or t'other
043 * </pre>
044 * <p/>
045 * <p>The syntax is roughly:</p>
046 * <p/>
047 * <pre>
048 * file      ::= parameter* section* EOF
049 * section   ::= header parameter*
050 * header    ::= '[' NAME ']'
051 * parameter ::= NAME '=' VALUE EOL
052 * </pre>
053 * <p/>
054 * <p>Blank lines, and lines that begin with either '#' or ';' are
055 * ignored. long lines may be continued by preceding the end-of-line
056 * character(s) with a backslash ('\').</p>
057 *
058 * @version $Revision$
059 */
060public final class Parameters
061{
062
063    // Constants and fields.
064    // -----------------------------------------------------------------------
065
066    /**
067     * The callback object.
068     */
069    private final ParameterListener listener;
070
071    /**
072     * The current file being read.
073     */
074    private LineNumberReader in;
075
076    // Constructor.
077    // -----------------------------------------------------------------------
078
079    /**
080     * Create a new parameter file parser. The argument is a concrete
081     * imlpmentation of {@link ParameterListener} which will take the
082     * parsed arguments.
083     *
084     * @param listener The parameter listener.
085     */
086    public Parameters(ParameterListener listener)
087    {
088        this.listener = listener;
089    }
090
091    // Instance methods.
092    // -----------------------------------------------------------------------
093
094    /**
095     * Begin parsing file <i>filename</i>.
096     *
097     * @param filename The name of the file to parse.
098     * @throws IOException If an I/O error occurs.
099     */
100    public void begin(String filename) throws IOException
101    {
102        in = new LineNumberReader(new FileReader(filename));
103    }
104
105    /**
106     * Parse, or continue parsing the file if a parsing error occured
107     * in a previous call to this method. A call to {@link
108     * #begin(java.lang.String)} must have succeeded before this method
109     * is called.
110     *
111     * @throws ParameterException If a parsing error occurs. Parsing can
112     *                            continue if this exception is thrown by calling this
113     *                            method again.
114     * @throws IOException        If an I/O error occurs.
115     */
116    public void parse() throws IOException
117    {
118        if (in == null)
119        {
120            throw new IOException("nothing to parse");
121        }
122        String line;
123
124        while ((line = in.readLine()) != null)
125        {
126            if (isIgnorable(line)) continue;
127
128            // Concatenate continuation lines.
129            while (line.endsWith("\\"))
130            {
131                String line2 = in.readLine();
132                if (line2 == null) break;
133                if (isIgnorable(line2)) continue;
134                line = line.substring(0, line.length() - 1);
135
136                // Make sure we aren't fooled by '\' followed by a space.
137                if (line2.endsWith("\\"))
138                {
139                    line += line2.trim();
140                } else if (line2.trim().endsWith("\\"))
141                {
142                    line2 = line2.trim() + " ";
143                    line += line2;
144                } else
145                {
146                    line += line2.trim();
147                }
148            }
149
150            line = line.trim();
151            int i;
152            if (line.startsWith("[") && line.endsWith("]"))
153            {
154                listener.beginSection(crush(line.substring(1, line.length() - 1)));
155            } else if ((i = line.indexOf('=')) > 1)
156            {
157                listener.setParameter(crush(line.substring(0, i).trim()),
158                        crush(line.substring(i + 1).trim()));
159            } else
160            {
161                throw new ParameterException("malformed line at "
162                        + in.getLineNumber());
163            }
164        }
165
166        in.close();
167        in = null;
168    }
169
170    // Own methods.
171    // -----------------------------------------------------------------------
172
173    /**
174     * Test if a line is a comment or empty.
175     *
176     * @param s The string to test.
177     * @return true if this string is a comment or empty.
178     */
179    private static boolean isIgnorable(String s)
180    {
181        s = s.trim();
182        if (s.length() == 0) return true;
183        if (s.charAt(0) == '#' || s.charAt(0) == ';') return true;
184        return false;
185    }
186
187    /**
188     * Crush multiple spaces (or other whitespace) into a single space.
189     *
190     * @param s The string to crush.
191     * @return The crushed string.
192     */
193    private static String crush(String s)
194    {
195        StringTokenizer tok = new StringTokenizer(s);
196        if (tok.countTokens() <= 1) return s;
197        String result = tok.nextToken();
198        while (tok.hasMoreTokens())
199            result += " " + tok.nextToken();
200        return result;
201    }
202}

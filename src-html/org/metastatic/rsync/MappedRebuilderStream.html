023package org.metastatic.rsync;
024
025import java.io.File;
026import java.io.IOException;
027
028import java.nio.MappedByteBuffer;
029import java.nio.channels.FileChannel;
030
031import java.util.Iterator;
032
033/**
034 * A version of {@link RebuilderStream} that maps large files to memory
035 * using the NIO API. Small files are not mapped and automatically use
036 * the superclass's implementation.
037 */
038public class MappedRebuilderStream extends RebuilderStream
039{
040
041    // Constants and fields.
042    // -----------------------------------------------------------------------
043
044    /**
045     * The default lower bound for files to map.
046     */
047    public static final int MAP_LIMIT = 32768;
048
049    /**
050     * The size of the map. If not specified, the entire file is mapped.
051     */
052    protected long mapSize;
053
054    /**
055     * The lower bound file length to map; files smaller than this will
056     * not be mapped.
057     */
058    protected long mapLimit;
059
060    /**
061     * The mapped file, if any.
062     */
063    protected MappedByteBuffer mappedFile;
064
065    /**
066     * The current offset in the file where the region is mapped.
067     */
068    protected long mapOffset;
069
070    // Constructors.
071    // -----------------------------------------------------------------------
072
073    /**
074     * Create a new memory mapped rebuilder, with the default map limit
075     * and a maximum map size of {@link java.lang.Integer#MAX_VALUE}.
076     */
077    public MappedRebuilderStream()
078    {
079        this(Integer.MAX_VALUE, MAP_LIMIT);
080    }
081
082    /**
083     * Create a new memory mapped rebuilder with the given map limit and
084     * a maximum map size of {@link java.lang.Integer#MAX_VALUE}.
085     *
086     * @param mapLimit The smallest file size to map.
087     */
088    public MappedRebuilderStream(long mapLimit)
089    {
090        this(Integer.MAX_VALUE, mapLimit);
091    }
092
093    /**
094     * Create a new memory mapped rebuilder with the given map limit and
095     * maximum map size.
096     *
097     * @param mapSize  The maximum size of map to create.
098     * @param mapLimit The smallest file size to map.
099     */
100    public MappedRebuilderStream(long mapSize, long mapLimit)
101    {
102        super();
103        this.mapSize = mapSize;
104        this.mapLimit = mapLimit;
105    }
106
107    // Instance methods.
108    // -----------------------------------------------------------------------
109
110    public void setBasisFile(File file) throws IOException
111    {
112        super.setBasisFile(file);
113        mappedFile = null;
114        if (file != null && file.length() >= mapLimit)
115        {
116            mappedFile = basisFile.getChannel().map(FileChannel.MapMode.READ_ONLY,
117                    0, Math.min(mapSize, file.length()));
118            mapOffset = 0;
119        }
120    }
121
122    public void setBasisFile(String filename) throws IOException
123    {
124        super.setBasisFile(filename);
125        mappedFile = null;
126        if (basisFile != null && basisFile.length() >= mapLimit)
127        {
128            mappedFile = basisFile.getChannel().map(FileChannel.MapMode.READ_ONLY,
129                    0, Math.min(mapSize, basisFile.length()));
130            mapOffset = 0;
131        }
132    }
133
134    public void update(Delta delta) throws IOException, ListenerException
135    {
136        if (mappedFile == null)
137        {
138            super.update(delta);
139        } else
140        {
141            RebuilderEvent event = null;
142            if (delta instanceof DataBlock)
143            {
144                event = new RebuilderEvent(((DataBlock) delta).getData(),
145                        delta.getWriteOffset());
146            } else
147            {
148                long offset = ((Offsets) delta).getOldOffset();
149                if (offset + delta.getBlockLength() > mapOffset + mappedFile.capacity())
150                    remapFile(offset);
151                byte[] buf = new byte[delta.getBlockLength()];
152                mappedFile.position((int) (offset - mapOffset));
153                mappedFile.get(buf);
154                event = new RebuilderEvent(buf, delta.getWriteOffset());
155            }
156            for (Iterator i = listeners.iterator(); i.hasNext(); )
157                ((RebuilderListener) i.next()).update(event);
158        }
159    }
160
161    // Own methods.
162    // -----------------------------------------------------------------------
163
164    /**
165     * Remap the file, if the read offset is not currently mapped.
166     *
167     * @param newOffset The new offset that needs to be read. The mapped
168     *                  region will have this offset in the middle of the buffer.
169     */
170    private void remapFile(long newOffset) throws IOException
171    {
172        long newLen = Math.min(mapSize, 2 * (basisFile.length() - newOffset));
173        mapOffset = newOffset - (newLen / 2);
174        if (mapOffset < 0)
175            mapOffset = 0;
176        mappedFile = basisFile.getChannel().map(
177                FileChannel.MapMode.READ_ONLY, mapOffset, newLen);
178    }
179}

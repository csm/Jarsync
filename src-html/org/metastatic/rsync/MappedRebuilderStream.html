044package org.metastatic.rsync;
045
046import java.io.File;
047import java.io.IOException;
048
049import java.nio.MappedByteBuffer;
050import java.nio.channels.FileChannel;
051
052import java.util.Iterator;
053
054/**
055 * A version of {@link RebuilderStream} that maps large files to memory
056 * using the NIO API. Small files are not mapped and automatically use
057 * the superclass's implementation.
058 */
059public class MappedRebuilderStream extends RebuilderStream
060{
061
062    // Constants and fields.
063    // -----------------------------------------------------------------------
064
065    /**
066     * The default lower bound for files to map.
067     */
068    public static final int MAP_LIMIT = 32768;
069
070    /**
071     * The size of the map. If not specified, the entire file is mapped.
072     */
073    protected long mapSize;
074
075    /**
076     * The lower bound file length to map; files smaller than this will
077     * not be mapped.
078     */
079    protected long mapLimit;
080
081    /**
082     * The mapped file, if any.
083     */
084    protected MappedByteBuffer mappedFile;
085
086    /**
087     * The current offset in the file where the region is mapped.
088     */
089    protected long mapOffset;
090
091    // Constructors.
092    // -----------------------------------------------------------------------
093
094    /**
095     * Create a new memory mapped rebuilder, with the default map limit
096     * and a maximum map size of {@link java.lang.Integer#MAX_VALUE}.
097     */
098    public MappedRebuilderStream()
099    {
100        this(Integer.MAX_VALUE, MAP_LIMIT);
101    }
102
103    /**
104     * Create a new memory mapped rebuilder with the given map limit and
105     * a maximum map size of {@link java.lang.Integer#MAX_VALUE}.
106     *
107     * @param mapLimit The smallest file size to map.
108     */
109    public MappedRebuilderStream(long mapLimit)
110    {
111        this(Integer.MAX_VALUE, mapLimit);
112    }
113
114    /**
115     * Create a new memory mapped rebuilder with the given map limit and
116     * maximum map size.
117     *
118     * @param mapSize  The maximum size of map to create.
119     * @param mapLimit The smallest file size to map.
120     */
121    public MappedRebuilderStream(long mapSize, long mapLimit)
122    {
123        super();
124        this.mapSize = mapSize;
125        this.mapLimit = mapLimit;
126    }
127
128    // Instance methods.
129    // -----------------------------------------------------------------------
130
131    public void setBasisFile(File file) throws IOException
132    {
133        super.setBasisFile(file);
134        mappedFile = null;
135        if (file != null && file.length() >= mapLimit)
136        {
137            mappedFile = basisFile.getChannel().map(FileChannel.MapMode.READ_ONLY,
138                    0, Math.min(mapSize, file.length()));
139            mapOffset = 0;
140        }
141    }
142
143    public void setBasisFile(String filename) throws IOException
144    {
145        super.setBasisFile(filename);
146        mappedFile = null;
147        if (basisFile != null && basisFile.length() >= mapLimit)
148        {
149            mappedFile = basisFile.getChannel().map(FileChannel.MapMode.READ_ONLY,
150                    0, Math.min(mapSize, basisFile.length()));
151            mapOffset = 0;
152        }
153    }
154
155    public void update(Delta delta) throws IOException, ListenerException
156    {
157        if (mappedFile == null)
158        {
159            super.update(delta);
160        } else
161        {
162            RebuilderEvent event = null;
163            if (delta instanceof DataBlock)
164            {
165                event = new RebuilderEvent(((DataBlock) delta).getData(),
166                        delta.getWriteOffset());
167            } else
168            {
169                long offset = ((Offsets) delta).getOldOffset();
170                if (offset + delta.getBlockLength() > mapOffset + mappedFile.capacity())
171                    remapFile(offset);
172                byte[] buf = new byte[delta.getBlockLength()];
173                mappedFile.position((int) (offset - mapOffset));
174                mappedFile.get(buf);
175                event = new RebuilderEvent(buf, delta.getWriteOffset());
176            }
177            for (Iterator i = listeners.iterator(); i.hasNext(); )
178                ((RebuilderListener) i.next()).update(event);
179        }
180    }
181
182    // Own methods.
183    // -----------------------------------------------------------------------
184
185    /**
186     * Remap the file, if the read offset is not currently mapped.
187     *
188     * @param newOffset The new offset that needs to be read. The mapped
189     *                  region will have this offset in the middle of the buffer.
190     */
191    private void remapFile(long newOffset) throws IOException
192    {
193        long newLen = Math.min(mapSize, 2 * (basisFile.length() - newOffset));
194        mapOffset = newOffset - (newLen / 2);
195        if (mapOffset < 0)
196            mapOffset = 0;
197        mappedFile = basisFile.getChannel().map(
198                FileChannel.MapMode.READ_ONLY, mapOffset, newLen);
199    }
200}

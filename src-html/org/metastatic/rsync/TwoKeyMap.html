050/**
051 * <p>This is a "double-keyed" mapping. The first key is a 16-bit integer,
052 * and the second key is a variable-length byte array. With this, we can
053 * compute if a given mapping is "probably" in the map using the first key,
054 * and compute whether or not a mapping is definitely in the hashtable with
055 * the second key. The rationale behind this is that the first key is
056 * trivial to compute and that the second key is more difficult to compute
057 * but more unique.</p>
058 *
059 * <p>Since the strong key can be a byte array of any length, then this
060 * "strong" key can be shorter (and thus less unique) than the "weak"
061 * key. For this class to work properly, the stronger key should be at
062 * least four bytes in length, preferably longer.</p>
063 *
064 * <p>The weak-key/strong-key method is inspired by (and is was written
065 * for) the "hashtable" in the rsync algorithm, and has three levels of
066 * key search:</p>
067 *
068 * <ol>
069 * <li>Test if the lower 2 bytes of the weak key (the positive part of a
070 * 32-bit integer in Java) have been mapped to anything yet. This method
071 * always takes O(1) time, and it is assumed that the weak key is
072 * trivial to compute.</li>
073 *
074 * <li>Test if the entire weak key is mapped to anything. Since this
075 * class uses a linked list to handle collisions where the lower 2 bytes
076 * are the same, this method takes at most O(n) operations (also
077 * assuming that the weak key is trivial to compute).</li>
078 *
079 * <li>Test if both the weak and strong keys map to an Object. In
080 * addition to the linked-list search of the second step, this involves
081 * a search of a red-black tree, meaning that the upper-bound time
082 * complexity ranges from O(lg(n)) to O(n). This works under the
083 * assumption that the strong key is some sort of {@link
084 * java.security.MessageDigest}, and thus takes longer to compute.</li>
085 * </ol>
086 *
087 * <p>With this method, we can determine if it is worth it to compute the
088 * strong key if we have already computed the weak key.</p>
089 *
090 * <p><code>null</code> is not a valid key in this map.</p>
091 *
092 * @author Casey Marshall
093 * @version $Revision$
094 */
095public class TwoKeyMap<T> implements java.io.Serializable, Map<ChecksumPair, T>
096{
097    // Constants and variables.
098    // -----------------------------------------------------------------
099
100    private final Map<Integer, Map<StrongKey, T>> map;
101
102    // Inner classes.
103    // -----------------------------------------------------------------
104
105    /**
106     * The stronger of the two keys in this {@link java.util.Map}. It is
107     * basically a wrapper around an array of bytes, providing methods
108     * important for using this class as the key in a Map, namely the
109     * {@link #hashCode()} and {@link #compareTo(java.lang.Object)}
110     * methods.
111     *
112     * @author Casey Marshall
113     * @version 1.0
114     * @since 1.1
115     */
116    public static class StrongKey implements java.io.Serializable, Comparable
117    {
118
119        // Constants and variables.
120        // --------------------------------------------------------------
121
122        /**
123         * The key itself. An array of some number of bytes.
124         *
125         * @since 1.0
126         */
127        protected byte[] key;
128
129        // Constructors.
130        // --------------------------------------------------------------
131
132        /**
133         * Create a new key with the specified bytes. <code>key</code> can be
134         * <code>null</code>.
135         *
136         * @param key The bytes that will make up this key.
137         * @since 1.0
138         */
139        StrongKey(byte[] key)
140        {
141            if (key != null)
142                this.key = key.clone();
143            else
144                this.key = key;
145        }
146
147        // Instance methods.
148        // --------------------------------------------------------------
149
150        /**
151         * Return the bytes that compose this Key.
152         *
153         * @return {@link #key}, the bytes that compose this key.
154         * @since 1.0
155         */
156        public byte[] getBytes()
157        {
158            if (key != null)
159                return key.clone();
160            return null;
161        }
162
163        /**
164         * Set the byte array that composes this key.
165         *
166         * @param key The bytes that will compose this key.
167         * @since 1.0
168         */
169        public void setBytes(byte[] key)
170        {
171            if (key != null)
172                this.key = (byte[]) key.clone();
173            else
174                this.key = key;
175        }
176
177        /**
178         * The length, in bytes, of this key.
179         *
180         * @return The length of this key in bytes.
181         * @since 1.0
182         */
183        public int length()
184        {
185            if (key != null)
186                return key.length;
187            return 0;
188        }
189
190        // Public instance methods overriding java.lang.Object -----------
191
192        /**
193         * Return a zero-padded hexadecimal string representing this key.
194         *
195         * @return A hexadecimal string of the bytes in {@link #key}.
196         * @since 1.0
197         */
198        public String toString()
199        {
200            if (key == null || key.length == 0)
201                return "nil";
202            return Util.toHexString(key);
203        }
204
205        /**
206         * The hash code for this key. This is defined as the XOR of all
207         * 32-bit blocks of the {@link #key} array.
208         *
209         * @return The hash code for this key.
210         * @since 1.0
211         */
212        public int hashCode()
213        {
214            if (key == null)
215                return 0;
216            int code = 0;
217            for (int i = key.length - 1; i >= 0; i--)
218                code ^= ((int) key[i] & 0xff) << (((key.length - i - 1) * 8) % 32);
219            return code;
220        }
221
222        /**
223         * Test if this key equals another. Two keys are equal if the method
224         * {@link java.util.Arrays#equals(byte[], byte[])} returns true for
225         * thier key arrays.
226         *
227         * @param o The object to compare to.
228         * @return <tt>true</tt> If this key is equivalent to the argument.
229         * @throws java.lang.ClassCastException If o is not a StrongKey.
230         * @since 1.0
231         */
232        public boolean equals(Object o)
233        {
234            return Arrays.equals(key, ((StrongKey) o).key);
235        }
236
237        // java.lang.Comparable interface implementation -----------------
238
239        /**
240         * Compare this object to another. This method returns an integer
241         * value less than, equal to, or greater than zero if this key
242         * is less than, equal to, or greater than the given key. This
243         * method will return
244         * <ul>
245         * <li>0 if the {@link #key} fields are references to the same
246         * array.
247         * <li>1 if {@link #key} in this class is null.
248         * <li>-1 if {@link #key} in <tt>o</tt> is null (null is always less
249         * than everything).
250         * <li>0 if the lengths of the {@link #key} arrays are the same and
251         * their contents are equivalent.
252         * <li>The difference between the lengths of the keys if different.
253         * <li>The difference between the first two different members of the
254         * arrays.
255         * </ul>
256         *
257         * @param o The key to compare to.
258         * @return An integer derived from the differences of the two keys.
259         * @throws java.lang.ClassCastException If o is not a StrongKey.
260         * @since 1.0
261         */
262        public int compareTo(Object o)
263        {
264            if (!(o instanceof StrongKey))
265                throw new ClassCastException(o.getClass().getName());
266            StrongKey sk = (StrongKey) o;
267            if (key == sk.key) return 0;
268            if (key == null) return 1;
269            if (sk.key == null) return -1;
270            if (java.util.Arrays.equals(key, sk.getBytes())) return 0;
271            if (key.length != sk.length()) return key.length - sk.length();
272            byte[] arr = sk.getBytes();
273            for (int i = 0; i < key.length; i++)
274                if (key[i] != arr[i]) return (key[i] - arr[i]);
275            return 0; // unreachable
276        }
277    }

293     * Create a new Map with all the mappings of the given map.
294     *
295     * @param m The initial mappings this Map should contain.
296     * @throws java.lang.ClassCastException   If one of the keys in
297     *                                        <code>m</code> is not a ChecksumPair.

303    {
304        this();

306    }

321    {
322        return map.containsKey(key);
323    }

325    /**
326     * Remove all entries from this map that are keyed by the given weak key.
327     *
328     * @param key The weak key to remove.

338    /**
339     * Put the given object at the location specified by a key pair.
340     *
341     * @param key   The {@link ChecksumPair} to use as the key.
342     * @param value The value to map to.
343     * @return The old value mapped, if any.
344     * @since 1.1
345     */
346    public T put(ChecksumPair key, T value)
347    {
348        Map<StrongKey, T> m = map.get(key.getWeak());
349        if (m == null)
350        {
351            m = new TreeMap<StrongKey, T>();
352            map.put(key.getWeak(), m);
353        }
354        StrongKey strongKey = new StrongKey(key.getStrong());
355        return m.put(strongKey, value);
356    }
357
358    /**
359     * Test if this map contains either the supplied weak key (if the
360     * argument is an Integer) or both the weak and strong keys (if the
361     * argument is a {@link ChecksumPair}).
362     *
363     * @param key The key to check.
364     * @return true If the map contains the given weak key (if
365     * <code>key</code> is an Integer) or the given pair of keys (if
366     * <code>key</code> is a {@link ChecksumPair}).
367     * @since 1.1
368     */
369    public boolean containsKey(Object key)
370    {
371        if (key instanceof Integer)
372        {
373            return containsKey(((Integer) key).intValue());
374        }
375        else if (key instanceof ChecksumPair)
376        {
377            Map<StrongKey, T> m = map.get(((ChecksumPair) key).getWeak());
378            if (m != null)
379            {
380                return m.containsKey(new StrongKey(((ChecksumPair) key).getStrong()));
381            }
382        }
383        return false;
384    }
385
386    /**
387     * Get the object mapped to by the given key pair. The argument
388     * SHOULD be a {@link ChecksumPair}.
389     *
390     * @param key The key of the object to get.
391     * @return The object keyed by <code>key</code>, or <code>null</code>
392     * if there is no such object.
393     * @since 1.1
394     */
395    public T get(Object key)
396    {
397        ChecksumPair pair = (ChecksumPair) key;
398        Map<StrongKey, T> m = map.get(pair.getWeak());
399        if (m != null)
400            return m.get(new StrongKey(pair.getStrong()));
401        return null;
402    }

414    /**
415     * Test if the given value is in one of the sub-tables.
416     *
417     * @param value The value to search for.
418     * @return <tt>true</tt> if <tt>value</tt> is in this Map.
419     * @since 1.1
420     */
421    public boolean containsValue(Object value)
422    {
423        for (Map<StrongKey, T> m : map.values())
424        {
425            if (m.containsValue(value))
426                return true;
427        }
428        return false;
429    }

431    /**
432     * Return an unmodifiable set of the SubTable objects in this class.
433     *
434     * @return A set of all sub-tables from this class.
435     * @since 1.1
436     */
437    public Set<Entry<ChecksumPair, T>> entrySet()
438    {
439        return new AbstractSet<Entry<ChecksumPair, T>>()
440        {
441            @Override
442            public Iterator<Entry<ChecksumPair, T>> iterator()
443            {
444                Iterator<Iterator<Entry<ChecksumPair, T>>> it =
445                        Iterators.transform(map.entrySet().iterator(), new Function<Entry<Integer, Map<StrongKey, T>>, Iterator<Entry<ChecksumPair, T>>>()
446                        {
447                            @Override
448                            public Iterator<Entry<ChecksumPair, T>> apply(final Entry<Integer, Map<StrongKey, T>> input)
449                            {
450                                final Iterator<Entry<StrongKey, T>> it = input.getValue().entrySet().iterator();
451                                return new Iterator<Entry<ChecksumPair, T>>()
452                                {
453                                    @Override
454                                    public boolean hasNext()
455                                    {
456                                        return it.hasNext();
457                                    }
458
459                                    @Override
460                                    public Entry<ChecksumPair, T> next()
461                                    {
462                                        Entry<StrongKey, T> next = it.next();
463                                        return new AbstractMap.SimpleEntry<ChecksumPair, T>(new ChecksumPair(input.getKey(), next.getKey().getBytes()), next.getValue());
464                                    }
465
466                                    @Override
467                                    public void remove()
468                                    {
469                                        it.remove();
470                                    }
471                                };
472                            }
473                        });
474                return Iterators.concat(it);
475            }
476
477            @Override
478            public int size()
479            {
480                return TwoKeyMap.this.size();
481            }
482        };
483    }

485    /**
486     * Test if this object equals another.
487     *
488     * @return <tt>true</tt> if <tt>o</tt> is an instance of this class
489     * and if it contains the same sub-tables.
490     * @throws java.lang.ClassCastException   if <tt>o</tt> is not an
491     *                                        instance of this class.
492     * @throws java.lang.NullPointerException if <tt>o</tt> is null.
493     * @since 1.1
494     */
495    public boolean equals(Object o)
496    {
497        try
498        {
499            Map<ChecksumPair, T> m = (Map<ChecksumPair, T>) o;
500            return entrySet().equals(m.entrySet());
501        }
502        catch (ClassCastException e)
503        {
504        }
505        catch (NullPointerException e)
506        {
507        }
508        return false;
509    }

533    /**
534     * Return an unmodifiable set of all the pairs of keys in this
535     * mapping.
536     *
537     * @return A set of all the {@link ChecksumPair}s in this mapping.
538     * @since 1.1
539     */
540    public Set<ChecksumPair> keySet()
541    {
542        return new AbstractSet<ChecksumPair>()
543        {
544            @Override
545            public Iterator<ChecksumPair> iterator()
546            {
547                Iterator<Entry<ChecksumPair, T>> it = entrySet().iterator();
548                return Iterators.transform(entrySet().iterator(), new Function<Entry<ChecksumPair, T>, ChecksumPair>()
549                {
550                    @Override
551                    public ChecksumPair apply(Entry<ChecksumPair, T> input)
552                    {
553                        return input.getKey();
554                    }
555                });
556            }
557
558            @Override
559            public int size()
560            {
561                return TwoKeyMap.this.size();
562            }
563        };
564    }

566    /**
567     * Put every entry in <code>m</code> in this map. This method will
568     * only work if the keys of <code>m</code> are of type {@link
569     * ChecksumPair}.
570     *
571     * @param m The mappings to put.
572     * @throws java.lang.ClassCastException   If every key in <code>m</code>
573     *                                        is not a ChecksumPair.
574     * @throws java.lang.NullPointerException If a key in <code>m</code>
575     *                                        is null.
576     * @see #put(org.metastatic.rsync.ChecksumPair, T)
577     * @since 1.1
578     */
579    public void putAll(Map<? extends ChecksumPair, ? extends T> m)
580    {
581        for (Entry<? extends ChecksumPair, ? extends T> e : m.entrySet())
582            put(e.getKey(), e.getValue());
583    }

585    /**
586     * Removes a key from this map. The key must be a {@link org.metastatic.rsync.ChecksumPair}.
587     *
588     * @param k The key of the object to be removed.
589     * @return The removed object, if such a mapping existed.
590     * <code>null</code> otherwise.
591     * @since 1.1
592     */
593    public T remove(Object k)
594    {
595        ChecksumPair key = (ChecksumPair) k;
596        if (key != null)
597        {
598            Map<StrongKey, T> m = map.get(key.getWeak());
599            if (m != null)
600            {
601                T ret = m.remove(new StrongKey(key.getStrong()));
602                if (m.isEmpty())
603                    map.remove(key.getWeak());
604                return ret;
605            }
606            return null;
607        }
608        return null;
609    }

612     * Return the number of mappings.
613     *
614     * @return The number of mappings.
615     * @since 1.1
616     */
617    public int size()
618    {
619        int size = 0;
620        for (Map<StrongKey, T> m : map.values())
621            size += m.size();
622        return size;
623    }

658    /**
659     * Create a printable version of this Map.
660     *
661     * @return A {@link java.lang.String} representing this object.
662     * @since 1.1
663     */
664    public String toString()
665    {
666        StringBuilder str = new StringBuilder("{");
667        boolean first = true;
668        for (Entry<ChecksumPair, T> e : entrySet())
669        {
670            if (!first)
671                str.append(", ");
672            str.append('(');
673            str.append(Integer.toHexString(e.getKey().getWeak()));
674            str.append(", ");
675            str.append(Util.toHexString(e.getKey().getStrong()));
676            str.append(") => ");
677            str.append(e.getValue());
678            first = false;
679        }
680        str.append("}");
681        return str.toString();
682    }

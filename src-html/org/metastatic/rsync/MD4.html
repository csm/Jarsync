047package org.metastatic.rsync;
048
049import java.security.DigestException;
050import java.security.MessageDigestSpi;
051
052/**
053 * <p>An implementation of Ron Rivest's MD4 message digest algorithm.
054 * MD4 was the precursor to the stronger MD5
055 * algorithm, and while not considered cryptograpically secure itself,
056 * MD4 is in use in various applications. It is slightly faster than
057 * MD5.</p>
058 * <p/>
059 * <p>This implementation is derived from the version of MD4 in <a
060 * href="http://www.gnu.org/software/gnu-crypto/">GNU Crypto.</p>
061 * <p/>
062 * <p>References:</p>
063 * <p/>
064 * <ol>
065 * <li>The <a href="http://www.ietf.org/rfc/rfc1320.txt">MD4</a> Message-
066 * Digest Algorithm.<br>
067 * R. Rivest.</li>
068 * </ol>
069 *
070 * @version $Revision$
071 */
072public class MD4 extends MessageDigestSpi implements Cloneable
073{
074
075    // Constants and variables.
076    // -----------------------------------------------------------------
077
078    /**
079     * An MD4 message digest is always 128-bits long, or 16 bytes.
080     */
081    public static final int DIGEST_LENGTH = 16;
082
083    /**
084     * The MD4 algorithm operates on 512-bit blocks, or 64 bytes.
085     */
086    public static final int BLOCK_LENGTH = 64;
087
088    protected static final int A = 0x67452301;
089    protected static final int B = 0xefcdab89;
090    protected static final int C = 0x98badcfe;
091    protected static final int D = 0x10325476;
092
093    /* The four chaining variables. */
094    protected int a, b, c, d;
095
096    protected long count;
097
098    protected final byte[] buffer;
099
100    /**
101     * Word buffer for transforming.
102     */
103    private final int[] X = new int[16];
104
105    // Constructors.
106    // -----------------------------------------------------------------
107
108    /**
109     * Trivial zero-argument constructor.
110     */
111    public MD4()
112    {
113        buffer = new byte[BLOCK_LENGTH];
114        engineReset();
115    }
116
117    /**
118     * Private constructor for cloning.
119     */
120    private MD4(MD4 that)
121    {
122        this();
123
124        this.a = that.a;
125        this.b = that.b;
126        this.c = that.c;
127        this.d = that.d;
128        this.count = that.count;
129        System.arraycopy(that.buffer, 0, this.buffer, 0, BLOCK_LENGTH);
130    }
131
132    // java.lang.Cloneable interface implementation --------------------
133
134    public Object clone()
135    {
136        return new MD4(this);
137    }
138
139    // SPI instance methods.
140    // -----------------------------------------------------------------
141
142    protected int engineGetDigestLength()
143    {
144        return DIGEST_LENGTH;
145    }
146
147    public void engineUpdate(byte b)
148    {
149        // compute number of bytes still unhashed; ie. present in buffer
150        int i = (int) (count % BLOCK_LENGTH);
151        count++;
152        buffer[i] = b;
153        if (i == (BLOCK_LENGTH - 1))
154        {
155            transform(buffer, 0);
156        }
157    }
158
159    protected void engineUpdate(byte[] b, int offset, int len)
160    {
161        int n = (int) (count % BLOCK_LENGTH);
162        count += len;
163        int partLen = BLOCK_LENGTH - n;
164        int i = 0;
165
166        if (len >= partLen)
167        {
168            System.arraycopy(b, offset, buffer, n, partLen);
169            transform(buffer, 0);
170            for (i = partLen; i + BLOCK_LENGTH - 1 < len; i += BLOCK_LENGTH)
171            {
172                transform(b, offset + i);
173            }
174            n = 0;
175        }
176
177        if (i < len)
178        {
179            System.arraycopy(b, offset + i, buffer, n, len - i);
180        }
181    }
182
183    /**
184     * Pack the four chaining variables into a byte array.
185     */
186    protected byte[] engineDigest()
187    {
188        byte[] tail = padBuffer();
189        engineUpdate(tail, 0, tail.length);
190        byte[] digest = {
191                (byte) a, (byte) (a >>> 8), (byte) (a >>> 16), (byte) (a >>> 24),
192                (byte) b, (byte) (b >>> 8), (byte) (b >>> 16), (byte) (b >>> 24),
193                (byte) c, (byte) (c >>> 8), (byte) (c >>> 16), (byte) (c >>> 24),
194                (byte) d, (byte) (d >>> 8), (byte) (d >>> 16), (byte) (d >>> 24)
195        };
196
197        engineReset();
198
199        return digest;
200    }
201
202    protected int engineDigest(byte[] out, int off, int len) throws DigestException
203    {
204        if (off < 0 || off + len >= out.length)
205        {
206            throw new DigestException();
207        }
208        System.arraycopy(engineDigest(), 0, out, off,
209                Math.min(len, DIGEST_LENGTH));
210        return Math.min(len, DIGEST_LENGTH);
211    }
212
213    /**
214     * Reset the four chaining variables.
215     */
216    protected void engineReset()
217    {
218        a = A;
219        b = B;
220        c = C;
221        d = D;
222        count = 0;
223    }
224
225    /**
226     * Pad the buffer by appending the byte 0x80, then as many zero bytes
227     * to fill the buffer 8 bytes shy of being a multiple of 64 bytes, then
228     * append the length of the buffer, in bits, before padding.
229     */
230    protected byte[] padBuffer()
231    {
232        int n = (int) (count % BLOCK_LENGTH);
233        int padding = (n < 56) ? (56 - n) : (120 - n);
234        byte[] pad = new byte[padding + 8];
235
236        pad[0] = (byte) 0x80;
237        long bits = count << 3;
238        pad[padding++] = (byte) bits;
239        pad[padding++] = (byte) (bits >>> 8);
240        pad[padding++] = (byte) (bits >>> 16);
241        pad[padding++] = (byte) (bits >>> 24);
242        pad[padding++] = (byte) (bits >>> 32);
243        pad[padding++] = (byte) (bits >>> 40);
244        pad[padding++] = (byte) (bits >>> 48);
245        pad[padding] = (byte) (bits >>> 56);
246
247        return pad;
248    }
249
250    /**
251     * Transform a 64-byte block.
252     */
253    protected void transform(byte[] in, int offset)
254    {
255        int aa, bb, cc, dd;
256
257        for (int i = 0, n = 0; i < 16; i++)
258        {
259            X[i] = (in[offset++] & 0xff) |
260                    (in[offset++] & 0xff) << 8 |
261                    (in[offset++] & 0xff) << 16 |
262                    (in[offset++] & 0xff) << 24;
263        }
264
265        aa = a;
266        bb = b;
267        cc = c;
268        dd = d;
269
270        // Round 1
271        a += ((b & c) | ((~b) & d)) + X[0];
272        a = a << 3 | a >>> (32 - 3);
273        d += ((a & b) | ((~a) & c)) + X[1];
274        d = d << 7 | d >>> (32 - 7);
275        c += ((d & a) | ((~d) & b)) + X[2];
276        c = c << 11 | c >>> (32 - 11);
277        b += ((c & d) | ((~c) & a)) + X[3];
278        b = b << 19 | b >>> (32 - 19);
279        a += ((b & c) | ((~b) & d)) + X[4];
280        a = a << 3 | a >>> (32 - 3);
281        d += ((a & b) | ((~a) & c)) + X[5];
282        d = d << 7 | d >>> (32 - 7);
283        c += ((d & a) | ((~d) & b)) + X[6];
284        c = c << 11 | c >>> (32 - 11);
285        b += ((c & d) | ((~c) & a)) + X[7];
286        b = b << 19 | b >>> (32 - 19);
287        a += ((b & c) | ((~b) & d)) + X[8];
288        a = a << 3 | a >>> (32 - 3);
289        d += ((a & b) | ((~a) & c)) + X[9];
290        d = d << 7 | d >>> (32 - 7);
291        c += ((d & a) | ((~d) & b)) + X[10];
292        c = c << 11 | c >>> (32 - 11);
293        b += ((c & d) | ((~c) & a)) + X[11];
294        b = b << 19 | b >>> (32 - 19);
295        a += ((b & c) | ((~b) & d)) + X[12];
296        a = a << 3 | a >>> (32 - 3);
297        d += ((a & b) | ((~a) & c)) + X[13];
298        d = d << 7 | d >>> (32 - 7);
299        c += ((d & a) | ((~d) & b)) + X[14];
300        c = c << 11 | c >>> (32 - 11);
301        b += ((c & d) | ((~c) & a)) + X[15];
302        b = b << 19 | b >>> (32 - 19);
303
304        // Round 2.
305        a += ((b & (c | d)) | (c & d)) + X[0] + 0x5a827999;
306        a = a << 3 | a >>> (32 - 3);
307        d += ((a & (b | c)) | (b & c)) + X[4] + 0x5a827999;
308        d = d << 5 | d >>> (32 - 5);
309        c += ((d & (a | b)) | (a & b)) + X[8] + 0x5a827999;
310        c = c << 9 | c >>> (32 - 9);
311        b += ((c & (d | a)) | (d & a)) + X[12] + 0x5a827999;
312        b = b << 13 | b >>> (32 - 13);
313        a += ((b & (c | d)) | (c & d)) + X[1] + 0x5a827999;
314        a = a << 3 | a >>> (32 - 3);
315        d += ((a & (b | c)) | (b & c)) + X[5] + 0x5a827999;
316        d = d << 5 | d >>> (32 - 5);
317        c += ((d & (a | b)) | (a & b)) + X[9] + 0x5a827999;
318        c = c << 9 | c >>> (32 - 9);
319        b += ((c & (d | a)) | (d & a)) + X[13] + 0x5a827999;
320        b = b << 13 | b >>> (32 - 13);
321        a += ((b & (c | d)) | (c & d)) + X[2] + 0x5a827999;
322        a = a << 3 | a >>> (32 - 3);
323        d += ((a & (b | c)) | (b & c)) + X[6] + 0x5a827999;
324        d = d << 5 | d >>> (32 - 5);
325        c += ((d & (a | b)) | (a & b)) + X[10] + 0x5a827999;
326        c = c << 9 | c >>> (32 - 9);
327        b += ((c & (d | a)) | (d & a)) + X[14] + 0x5a827999;
328        b = b << 13 | b >>> (32 - 13);
329        a += ((b & (c | d)) | (c & d)) + X[3] + 0x5a827999;
330        a = a << 3 | a >>> (32 - 3);
331        d += ((a & (b | c)) | (b & c)) + X[7] + 0x5a827999;
332        d = d << 5 | d >>> (32 - 5);
333        c += ((d & (a | b)) | (a & b)) + X[11] + 0x5a827999;
334        c = c << 9 | c >>> (32 - 9);
335        b += ((c & (d | a)) | (d & a)) + X[15] + 0x5a827999;
336        b = b << 13 | b >>> (32 - 13);
337
338        // Round 3.
339        a += (b ^ c ^ d) + X[0] + 0x6ed9eba1;
340        a = a << 3 | a >>> (32 - 3);
341        d += (a ^ b ^ c) + X[8] + 0x6ed9eba1;
342        d = d << 9 | d >>> (32 - 9);
343        c += (d ^ a ^ b) + X[4] + 0x6ed9eba1;
344        c = c << 11 | c >>> (32 - 11);
345        b += (c ^ d ^ a) + X[12] + 0x6ed9eba1;
346        b = b << 15 | b >>> (32 - 15);
347        a += (b ^ c ^ d) + X[2] + 0x6ed9eba1;
348        a = a << 3 | a >>> (32 - 3);
349        d += (a ^ b ^ c) + X[10] + 0x6ed9eba1;
350        d = d << 9 | d >>> (32 - 9);
351        c += (d ^ a ^ b) + X[6] + 0x6ed9eba1;
352        c = c << 11 | c >>> (32 - 11);
353        b += (c ^ d ^ a) + X[14] + 0x6ed9eba1;
354        b = b << 15 | b >>> (32 - 15);
355        a += (b ^ c ^ d) + X[1] + 0x6ed9eba1;
356        a = a << 3 | a >>> (32 - 3);
357        d += (a ^ b ^ c) + X[9] + 0x6ed9eba1;
358        d = d << 9 | d >>> (32 - 9);
359        c += (d ^ a ^ b) + X[5] + 0x6ed9eba1;
360        c = c << 11 | c >>> (32 - 11);
361        b += (c ^ d ^ a) + X[13] + 0x6ed9eba1;
362        b = b << 15 | b >>> (32 - 15);
363        a += (b ^ c ^ d) + X[3] + 0x6ed9eba1;
364        a = a << 3 | a >>> (32 - 3);
365        d += (a ^ b ^ c) + X[11] + 0x6ed9eba1;
366        d = d << 9 | d >>> (32 - 9);
367        c += (d ^ a ^ b) + X[7] + 0x6ed9eba1;
368        c = c << 11 | c >>> (32 - 11);
369        b += (c ^ d ^ a) + X[15] + 0x6ed9eba1;
370        b = b << 15 | b >>> (32 - 15);
371
372        a += aa;
373        b += bb;
374        c += cc;
375        d += dd;
376    }

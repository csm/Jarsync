024package org.metastatic.rsync;
025
026import java.io.IOException;
027import java.io.OutputStream;
028import java.lang.reflect.Constructor;
029import java.lang.reflect.InvocationTargetException;
030import java.util.List;
031
032/**
033 * The superclass of objects that encode sets of deltas to external
034 * representations, such as the over-the-wire format of rsync or the
035 * rdiff file format.
036 * <p/>
037 * <p>Subclasses MAY define themselves to be accessable through the
038 * {@link #getInstance(java.lang.String, org.metastatic.rsync.Configuration, java.io.OutputStream)} method
039 * by providing a one-argument constructor that accepts an {@link
040 * OutputStream} and defining the system property
041 * "jarsync.deltaEncoder.<i>encoding-name</i>".
042 */
043public abstract class DeltaEncoder
044{
045
046    // Constants and fields.
047    // -------------------------------------------------------------------------
048
049    public static final String PROPERTY = "jarsync.deltaEncoder.";
050
051    /**
052     * The configuration.
053     */
054    protected Configuration config;
055
056    /**
057     * The output stream.
058     */
059    protected OutputStream out;
060
061    // Constructors.
062    // -------------------------------------------------------------------------
063
064    /**
065     * Creates a new delta encoder.
066     *
067     * @param config The configuration.
068     * @param out    The output stream to write the data to.
069     */
070    public DeltaEncoder(Configuration config, OutputStream out)
071    {
072        this.config = (Configuration) config.clone();
073        this.out = out;
074    }
075
076    // Class methods.
077    // -------------------------------------------------------------------------
078
079    /**
080     * Returns a new instance of the specified encoder.
081     *
082     * @throws IllegalArgumentException If there is no appropriate
083     *                                  encoder available.
084     */
085    public static final DeltaEncoder getInstance(String encoding,
086                                                 Configuration config,
087                                                 OutputStream out)
088    {
089        if (encoding == null || config == null || out == null)
090            throw new NullPointerException();
091        if (encoding.length() == 0)
092            throw new IllegalArgumentException();
093        try
094        {
095            Class clazz = Class.forName(System.getProperty(PROPERTY + encoding));
096            if (!DeltaEncoder.class.isAssignableFrom(clazz))
097                throw new IllegalArgumentException(clazz.getName() +
098                        ": not a subclass of " +
099                        DeltaEncoder.class.getName());
100            Constructor c = clazz.getConstructor(new Class[]{Configuration.class,
101                    OutputStream.class});
102            return (DeltaEncoder) c.newInstance(new Object[]{config, out});
103        } catch (ClassNotFoundException cnfe)
104        {
105            throw new IllegalArgumentException("class not found: " +
106                    cnfe.getMessage());
107        } catch (NoSuchMethodException nsme)
108        {
109            throw new IllegalArgumentException("subclass has no constructor");
110        } catch (InvocationTargetException ite)
111        {
112            throw new IllegalArgumentException(ite.getMessage());
113        } catch (InstantiationException ie)
114        {
115            throw new IllegalArgumentException(ie.getMessage());
116        } catch (IllegalAccessException iae)
117        {
118            throw new IllegalArgumentException(iae.getMessage());
119        }
120    }
121
122    // Instance methods.
123    // -------------------------------------------------------------------------
124
125    /**
126     * Write (encode) a list of deltas to the output stream. This method does
127     * <b>not</b> call {@link #doFinal()}.
128     * <p/>
129     * <p>This method checks every element of the supplied list to ensure that
130     * all are either non-null or implement the {@link Delta} interface, before
131     * writing any data.
132     *
133     * @param deltas The list of deltas to write.
134     * @throws IOException              If an I/O error occurs.
135     * @throws IllegalArgumentException If any element of the list is not
136     *                                  a {@link Delta}.
137     * @throws NullPointerException     If any element is null.
138     */
139    public void write(List<Delta> deltas) throws IOException
140    {
141        for (Delta o : deltas)
142        {
143            if (o == null)
144                throw new NullPointerException();
145        }
146        for (Delta delta : deltas) write(delta);
147    }
148
149    // Abstract methods.
150    // -----------------------------------------------------------------------
151
152    /**
153     * Write (encode) a single delta to the output stream.
154     *
155     * @param d The delta to write.
156     * @throws IOException If an I/O error occurs.
157     */
158    public abstract void write(Delta d) throws IOException;
159
160    /**
161     * Finish encoding the deltas (at least, this set of deltas) and write any
162     * encoding-specific end-of-deltas entity.
163     *
164     * @throws IOException If an I/O error occurs.
165     */
166    public abstract void doFinal() throws IOException;
167
168    /**
169     * Returns whether or not this encoder requires the deltas it is
170     * presented to be in <i>write offset</i> order, that is, the deltas
171     * passed to the <code>write</code> methods <b>must</b> be presented
172     * in increasing order of their {@link Delta#getWriteOffset()}
173     * values.
174     *
175     * @return True if this encoder requires write order.
176     */
177    public abstract boolean requiresOrder();
178}

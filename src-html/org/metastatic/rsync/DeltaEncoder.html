043package org.metastatic.rsync;
044
045import java.io.IOException;
046import java.io.OutputStream;
047
048import java.lang.reflect.Constructor;
049import java.lang.reflect.InvocationTargetException;
050
051import java.util.Iterator;
052import java.util.List;
053
054/**
055 * The superclass of objects that encode sets of deltas to external
056 * representations, such as the over-the-wire format of rsync or the
057 * rdiff file format.
058 * <p/>
059 * <p>Subclasses MAY define themselves to be accessable through the
060 * {@link #getInstance(java.lang.String, org.metastatic.rsync.Configuration, java.io.OutputStream)} method
061 * by providing a one-argument constructor that accepts an {@link
062 * OutputStream} and defining the system property
063 * "jarsync.deltaEncoder.<i>encoding-name</i>".
064 */
065public abstract class DeltaEncoder
066{
067
068    // Constants and fields.
069    // -------------------------------------------------------------------------
070
071    public static final String PROPERTY = "jarsync.deltaEncoder.";
072
073    /**
074     * The configuration.
075     */
076    protected Configuration config;
077
078    /**
079     * The output stream.
080     */
081    protected OutputStream out;
082
083    // Constructors.
084    // -------------------------------------------------------------------------
085
086    /**
087     * Creates a new delta encoder.
088     *
089     * @param config The configuration.
090     * @param out    The output stream to write the data to.
091     */
092    public DeltaEncoder(Configuration config, OutputStream out)
093    {
094        this.config = (Configuration) config.clone();
095        this.out = out;
096    }
097
098    // Class methods.
099    // -------------------------------------------------------------------------
100
101    /**
102     * Returns a new instance of the specified encoder.
103     *
104     * @throws IllegalArgumentException If there is no appropriate
105     *                                  encoder available.
106     */
107    public static final DeltaEncoder getInstance(String encoding,
108                                                 Configuration config,
109                                                 OutputStream out)
110    {
111        if (encoding == null || config == null || out == null)
112            throw new NullPointerException();
113        if (encoding.length() == 0)
114            throw new IllegalArgumentException();
115        try
116        {
117            Class clazz = Class.forName(System.getProperty(PROPERTY + encoding));
118            if (!DeltaEncoder.class.isAssignableFrom(clazz))
119                throw new IllegalArgumentException(clazz.getName() +
120                        ": not a subclass of " +
121                        DeltaEncoder.class.getName());
122            Constructor c = clazz.getConstructor(new Class[]{Configuration.class,
123                    OutputStream.class});
124            return (DeltaEncoder) c.newInstance(new Object[]{config, out});
125        } catch (ClassNotFoundException cnfe)
126        {
127            throw new IllegalArgumentException("class not found: " +
128                    cnfe.getMessage());
129        } catch (NoSuchMethodException nsme)
130        {
131            throw new IllegalArgumentException("subclass has no constructor");
132        } catch (InvocationTargetException ite)
133        {
134            throw new IllegalArgumentException(ite.getMessage());
135        } catch (InstantiationException ie)
136        {
137            throw new IllegalArgumentException(ie.getMessage());
138        } catch (IllegalAccessException iae)
139        {
140            throw new IllegalArgumentException(iae.getMessage());
141        }
142    }
143
144    // Instance methods.
145    // -------------------------------------------------------------------------
146
147    /**
148     * Write (encode) a list of deltas to the output stream. This method does
149     * <b>not</b> call {@link #doFinal()}.
150     * <p/>
151     * <p>This method checks every element of the supplied list to ensure that
152     * all are either non-null or implement the {@link Delta} interface, before
153     * writing any data.
154     *
155     * @param deltas The list of deltas to write.
156     * @throws IOException              If an I/O error occurs.
157     * @throws IllegalArgumentException If any element of the list is not
158     *                                  a {@link Delta}.
159     * @throws NullPointerException     If any element is null.
160     */
161    public void write(List<Delta> deltas) throws IOException
162    {
163        for (Delta o : deltas)
164        {
165            if (o == null)
166                throw new NullPointerException();
167        }
168        for (Delta delta : deltas) write(delta);
169    }
170
171    // Abstract methods.
172    // -----------------------------------------------------------------------
173
174    /**
175     * Write (encode) a single delta to the output stream.
176     *
177     * @param d The delta to write.
178     * @throws IOException If an I/O error occurs.
179     */
180    public abstract void write(Delta d) throws IOException;
181
182    /**
183     * Finish encoding the deltas (at least, this set of deltas) and write any
184     * encoding-specific end-of-deltas entity.
185     *
186     * @throws IOException If an I/O error occurs.
187     */
188    public abstract void doFinal() throws IOException;
189
190    /**
191     * Returns whether or not this encoder requires the deltas it is
192     * presented to be in <i>write offset</i> order, that is, the deltas
193     * passed to the <code>write</code> methods <b>must</b> be presented
194     * in increasing order of their {@link Delta#getWriteOffset()}
195     * values.
196     *
197     * @return True if this encoder requires write order.
198     */
199    public abstract boolean requiresOrder();
200}

078    /**
079     * Reconstruct data into a new byte array.
080     *
081     * @param buf    The original data.
082     * @param deltas The deltas to apply.
083     * @return The reconstructed data.
084     */
085    public static byte[] rebuild(byte[] buf, List deltas)
086    {
087        ByteArrayOutputStream baos = new ByteArrayOutputStream();
088        try
089        {
090            rebuild(baos, buf, deltas);
091        } catch (IOException shouldNotHappen)
092        {
093        }
094        return baos.toByteArray();
095    }
096
097    /**
098     * Reconstruct data into an output stream.
099     *
100     * @param out    The sink for reconstructed data.
101     * @param buf    The original data.
102     * @param deltas The deltas to apply.
103     */
104    public static void rebuild(OutputStream out, byte[] buf, List<Delta> deltas)
105            throws IOException
106    {
107        Delta[] darray = deltas.toArray(new Delta[0]);
108        Arrays.sort(darray, new OffsetComparator());
109
110        for (Delta aDarray : darray)
111        {
112            if (aDarray instanceof DataBlock)
113            {
114                out.write(((DataBlock) aDarray).getData());
115            } else
116            {
117                out.write(buf, (int) ((Offsets) aDarray).getOldOffset(),
118                        aDarray.getBlockLength());
119            }
120        }
121    }

123    /**
124     * Reconstruct a file into an output stream.
125     *
126     * @param out     The sink for reconstructed data.
127     * @param oldFile The original file.
128     * @param deltas  The deltas to apply.
129     */
130    public static void
131    rebuild(OutputStream out, File oldFile, List<Delta> deltas)
132            throws IOException
133    {
134        RandomAccessFile f = new RandomAccessFile(oldFile, "r");
135        Delta[] darray = deltas.toArray(new Delta[0]);
136        Arrays.sort(darray, new OffsetComparator());
137        byte[] buf = new byte[1024];
138
139        for (int i = 0; i < darray.length; i++)
140        {
141            if (darray[i] instanceof DataBlock)
142            {
143                out.write(((DataBlock) darray[i]).getData());
144            } else
145            {
146                f.seek(((Offsets) darray[i]).getOldOffset());
147                int len = 0, total = 0;
148                do
149                {
150                    len = f.read(buf);
151                    total += len;
152                    out.write(buf, 0, len);
153                } while (total < darray[i].getBlockLength());
154            }
155        }
156    }

158    /**
159     * Reconstruct a file into a new file created with {@link
160     * java.io.File#createTempFile(java.lang.String, java.lang.String, java.io.File)}.
161     * This file can then be renamed to the destination.
162     *
163     * @param oldFile The original file.
164     * @param deltas  A collection of {@link Delta}s to apply to the original
165     *                file.
166     * @return A unique {@link java.io.File} containing the reconstruction.
167     */
168    public static File rebuildFile(File oldFile, List deltas)
169            throws IOException
170    {
171        File newFile = File.createTempFile(TMP_PREFIX, TMP_SUFFIX,
172                oldFile.getParentFile());
173        rebuildFile(oldFile, newFile, deltas);
174        return newFile;
175    }

177    /**
178     * Reconstruct a file into <code>newFile</code>.
179     *
180     * @param oldFile The original file.
181     * @param newFile The file to write the reconstruction to. This must be a
182     *                different file than <code>oldFile</code>
183     * @param deltas  The {@link Delta}s to apply.
184     */
185    public static void rebuildFile(File oldFile, File newFile, List<Delta> deltas)
186            throws IOException
187    {
188        if (oldFile.equals(newFile))
189        {
190            throw new IOException("cannot read and write to the same file");
191        }
192        RandomAccessFile out = new RandomAccessFile(newFile, "rw");
193        RandomAccessFile in = null;
194        try
195        {
196            in = new RandomAccessFile(oldFile, "r");
197        } catch (IOException ignore)
198        {
199        }
200
201        for (Delta o : deltas)
202        {
203            if (o instanceof DataBlock)
204            {
205                long off = ((DataBlock) o).getOffset();
206                out.seek(off);
207                out.write(((DataBlock) o).getData());
208            } else if (o instanceof Offsets)
209            {
210                if (in == null)
211                {
212                    throw new IOException("original file does not exist or not readable");
213                }
214                int len = ((Offsets) o).getBlockLength();
215                long off1 = ((Offsets) o).getOldOffset();
216                long off2 = ((Offsets) o).getNewOffset();
217                byte[] buf = new byte[len];
218                in.seek(off1);
219                in.read(buf);
220                out.seek(off2);
221                out.write(buf);
222            }
223        }
224
225        if (in != null)
226            in.close();
227        out.close();
228    }

230    /**
231     * Reconstruct a file in-place. The contents of <code>file</code> will be
232     * overwritten with the contents of the reconstructed file.
233     *
234     * @param file   The file to reconstruct.
235     * @param deltas The {@link Delta}s to apply.
236     */
237    public static void rebuildFileInPlace(File file, List<Delta> deltas)
238            throws IOException
239    {
240        final boolean copyOnly = !file.exists();
241        RandomAccessFile f = new RandomAccessFile(file, "rw");
242        List<Offsets> offsets = new LinkedList<Offsets>();
243        List<DataBlock> dataBlocks = new LinkedList<DataBlock>();
244        TreeMap<Offsets, Set<Offsets>> digraph = new TreeMap<Offsets, Set<Offsets>>(new OffsetComparator());
245        long newFileLength = 0;
246
247        for (Delta o : deltas)
248        {
249            if (o instanceof Offsets)
250            {
251                if (copyOnly)
252                    throw new IOException("original file does not exist.");
253                offsets.add((Offsets) o);
254                digraph.put((Offsets) o, new HashSet());
255                newFileLength = Math.max(newFileLength,
256                        ((Offsets) o).getNewOffset() + ((Offsets) o).getBlockLength());
257            } else if (o instanceof DataBlock)
258            {
259                dataBlocks.add((DataBlock) o);
260                newFileLength = Math.max(newFileLength,
261                        ((DataBlock) o).getOffset() + ((DataBlock) o).getBlockLength());
262            }
263        }
264
265        if (copyOnly)
266        {
267            for (Iterator i = dataBlocks.iterator(); i.hasNext(); )
268            {
269                DataBlock d = (DataBlock) i.next();
270                f.seek(d.getWriteOffset());
271                f.write(d.getData());
272            }
273            if (f.length() < newFileLength)
274            {
275                f.setLength(newFileLength);
276            }
277            f.close();
278            return;
279        }
280
281        // build the digraph
282        for (Iterator i = offsets.iterator(); i.hasNext(); )
283        {
284            Offsets o1 = (Offsets) i.next();
285            Set<Offsets> adj = digraph.get(o1);
286            for (Iterator j = offsets.iterator(); j.hasNext(); )
287            {
288                Offsets o2 = (Offsets) j.next();
289                if (o1 == o2) continue;
290                if (conflict(o1, o2))
291                {
292                    adj.add(o2);
293                }
294            }
295        }
296
297        // Sort the digraph topologically, removing nodes that cause cycles.
298        TopologicalSorter ts = new TopologicalSorter(digraph);
299        ts.sort();
300
301        for (Object o1 : ts.getCycleNodes())
302        {
303            Offsets o = (Offsets) o1;
304            byte[] buf = new byte[o.getBlockLength()];
305            f.seek(o.getOldOffset());
306            f.read(buf);
307            dataBlocks.add(new DataBlock(o.getNewOffset(), buf));
308        }
309
310        for (Object o1 : ts.getFinished())
311        {
312            Offsets o = (Offsets) o1;
313            byte[] buf = new byte[o.getBlockLength()];
314            f.seek(o.getOldOffset());
315            f.read(buf);
316            f.seek(o.getNewOffset());
317            f.write(buf);
318        }
319
320        for (DataBlock db : dataBlocks)
321        {
322            f.seek(db.getOffset());
323            f.write(db.getData());
324        }
325        if (f.length() > newFileLength)
326        {
327            // Doesn't exist everywhere.
328            f.setLength(newFileLength);
329        }
330
331        f.close();
332    }

336    /**
337     * Test if the first offset will write to the reading area of the second
338     * offset.
339     */
340    private static boolean conflict(Offsets o1, Offsets o2)
341    {
342        return (o1.getNewOffset() >= o2.getOldOffset()
343                && o1.getNewOffset() <= o2.getOldOffset() + o2.getBlockLength())
344                || (o1.getNewOffset() + o1.getBlockLength() >= o2.getOldOffset()
345                && o1.getNewOffset() + o1.getBlockLength() <= o2.getOldOffset()
346                + o2.getBlockLength());
347    }

352    /**
353     * <p>This class topologically sorts a directed graph encoded as a mapping
354     * from vertices to sets of adjacent vertices. It is used when
355     * reconstructing a file in-place, where it is possible that Offsets
356     * objects overlap.</p>
357     * <p/>
358     * <p>Since a topological ordering of a directed graph is only possible in
359     * acyclic graphs, this sorter will remove any nodes that cause cycles,
360     * leaving the rest of the graph acyclic. After the graph is sorted the
361     * ordering can be obtained with {@link #getFinished()}, which is a {@link
362     * java.util.List} of nodes in their sorted order. Nodes that were part of
363     * cycles but have been removed can be retrieved with {@link
364     * #getCycleNodes()}, which returns a {@link java.util.Set} of nodes that
365     * were removed.</p>
366     * <p/>
367     * <p>Despite the terminology of "removed" nodes, the graph passed to the
368     * constructor is not modified.</p>
369     *
370     * @version 1.1
371     */
372    private static class TopologicalSorter
373    {
374        private static enum Color
375        {
376            White,
377            Gray,
378            Black
379        }
380
381        // Constants and variables.
382        // --------------------------------------------------------------------
383
384        /**
385         * The color for unvisited nodes.
386         */
387        private static final String WHITE = "white";
388
389        /**
390         * The color for visited nodes.
391         */
392        private static final String GRAY = "gray";
393
394        /**
395         * The color for finished nodes.
396         */
397        private static final String BLACK = "black";
398
399        /**
400         * The graph, encoded as a mapping between nodes and {@link
401         * java.util.Set}s of adjacent nodes.
402         */
403        private Map<Offsets, Set<Offsets>> graph;
404
405        /**
406         * A mapping between nodes and their color.
407         */
408        private Map<Offsets, Color> colors;
409
410        /**
411         * The ordered list of finished nodes.
412         */
413        private List<Offsets> finished;
414
415        /**
416         * The set of nodes that had to be removed from cycles.
417         */
418        private Set<Offsets> cycleNodes;
419
420        // Constructor.
421        // --------------------------------------------------------------------
422
423        /**
424         * Create a new topological sorter for the given graph.
425         *
426         * @param graph The graph to sort.
427         */
428        TopologicalSorter(Map<Offsets, Set<Offsets>> graph)
429        {
430            this.graph = graph;
431            colors = new HashMap<Offsets, Color>();
432            finished = new LinkedList<Offsets>();
433            cycleNodes = new HashSet<Offsets>();
434        }
435
436        // Instance methods.
437        // --------------------------------------------------------------------
438
439        /**
440         * Sort the graph. After this method returns the finished and removed
441         * nodes are available.
442         */
443        void sort()
444        {
445            DFS();
446        }
447
448        /**
449         * Return the ordered list of nodes.
450         *
451         * @return The ordered nodes.
452         */
453        List<Offsets> getFinished()
454        {
455            return finished;
456        }
457
458        /**
459         * Get the nodes that caused cycles.
460         *
461         * @return The removed nodes.
462         */
463        Set<Offsets> getCycleNodes()
464        {
465            return cycleNodes;
466        }
467
468        // Own methods.
469
470        /**
471         * <p>Depth-first search with removing nodes with back edges. As this
472         * method runs it places nodes into a list, which is a topological
473         * sorting of this graph. Any node encountered with a back-edge is
474         * put into the cycle-nodes list.</p>
475         * <p/>
476         * <p>References:</p>
477         * <ol>
478         * <li>T. Cormen, C. Leiserson, and R. Rivest: <i>Introduction to
479         * Algorithms</i>, pp. 477-487 (1990 The Massachusetts Institute of
480         * Technology).</li>
481         * </ol>
482         */
483        private void DFS()
484        {
485            for (Offsets offsets : graph.keySet())
486            {
487                colors.put(offsets, Color.White);
488            }
489            for (Offsets u : graph.keySet())
490            {
491                if (colors.get(u) == Color.White)
492                {
493                    DFSVisit(u);
494                }
495            }
496        }
497
498        /**
499         * DFS_VISIT. Visit a single node in a depth-first search.
500         *
501         * @param u The node to visit.
502         */
503        private void DFSVisit(Offsets u)
504        {
505            colors.put(u, Color.Gray);
506            for (Offsets v : graph.get(u))
507            {
508                if (colors.get(v) == Color.White)
509                {
510                    DFSVisit(v);
511                } else if (colors.get(v) == Color.Gray)
512                {
513                    cycleNodes.add(u);
514                }
515            }
516            colors.put(u, Color.Black);
517            if (!cycleNodes.contains(u))
518            {
519                finished.add(u);
520            }
521        }
522    }

524    /**
525     * Sort Offsets and DataBlocks objects by increasing write offset.
526     */
527    private static class OffsetComparator implements Comparator
528    {
529        public OffsetComparator()
530        {
531        }
532
533        public int compare(Object o1, Object o2)
534        {
535            long offset1 = 0;
536            long offset2 = 0;
537            if (o1 instanceof Delta)
538            {
539                offset1 = ((Delta) o1).getWriteOffset();
540            }
541            if (o2 instanceof Delta)
542            {
543                offset2 = ((Delta) o2).getWriteOffset();
544            }
545            return (int) (offset1 - offset2);
546        }
547
548        public boolean equals(Object o)
549        {
550            return (o instanceof OffsetComparator);
551        }
552    }

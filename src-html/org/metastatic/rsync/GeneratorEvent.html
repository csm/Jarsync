023package org.metastatic.rsync;
024
025import java.util.EventObject;
026
027/**
028 * Generator events are created whenever a checksum pair has been
029 * created.
030 *
031 * @see GeneratorListener
032 * @see GeneratorStream
033 */
034public class GeneratorEvent extends EventObject
035{
036
037    // Constructors.
038    // -----------------------------------------------------------------------
039
040    /**
041     * Create a new generator event.
042     *
043     * @param location The checksum location.
044     */
045    public GeneratorEvent(ChecksumLocation location)
046    {
047        super(location);
048    }
049
050    // Instance methods.
051    // -----------------------------------------------------------------------
052
053    /**
054     * Returns the source of this event, already cast to a ChecksumPair.
055     *
056     * @return The checksum pair.
057     */
058    public ChecksumLocation getChecksumLocation()
059    {
060        return (ChecksumLocation) source;
061    }
062}

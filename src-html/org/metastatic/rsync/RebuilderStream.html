042package org.metastatic.rsync;
043
044import java.io.File;
045import java.io.IOException;
046import java.io.RandomAccessFile;
047
048import java.util.Iterator;
049import java.util.LinkedList;
050
051/**
052 * A "streaming" alternative to {@link Rebuilder}. To use this class,
053 * create an intsance with a file argument representing the file being
054 * rebuilt. Then register one or more implementations of the {@link
055 * RebuilderListener} interface, which will write the data to the new
056 * file. Then call the {@link #update(Delta)} method for each {@link
057 * Delta} to be applied.
058 * <p/>
059 * <p>Note that unlike the {@link GeneratorStream} and {@link
060 * MatcherStream} classes this class does not need a {@link
061 * Configuration}, nor does it have any "doFinal" method -- it is
062 * completely stateless (except for the file) and the operations are
063 * finished when the last delta has been applied.
064 * <p/>
065 * <p>This class is optimal for situations where the deltas are coming
066 * in a stream over a communications link, and when it would be
067 * inefficient to wait until all deltas are received.
068 */
069public class RebuilderStream
070{
071
072    // Fields.
073    // -----------------------------------------------------------------------
074
075    /**
076     * The basis file.
077     */
078    protected RandomAccessFile basisFile;
079
080    /**
081     * The list of {@link RebuilderListener}s.
082     */
083    protected final LinkedList<RebuilderListener> listeners;
084
085    // Constructors.
086    // -----------------------------------------------------------------------
087
088    /**
089     * Create a new rebuilder.
090     */
091    public RebuilderStream()
092    {
093        listeners = new LinkedList<RebuilderListener>();
094    }
095
096    // Instance methods.
097    // -----------------------------------------------------------------------
098
099    /**
100     * Add a RebuilderListener listener to this rebuilder.
101     *
102     * @param listener The listener to add.
103     * @throws IllegalArgumentException If <i>listener</i> is null.
104     */
105    public void addListener(RebuilderListener listener)
106    {
107        if (listener == null)
108            throw new IllegalArgumentException();
109        listeners.add(listener);
110    }
111
112    /**
113     * Remove a listener from this rebuilder.
114     *
115     * @param listener The listener to remove.
116     */
117    public void removeListener(RebuilderListener listener)
118    {
119        listeners.remove(listener);
120    }
121
122    /**
123     * Set the basis file.
124     *
125     * @param file The basis file.
126     * @throws IOException If the file is not readable.
127     */
128    public void setBasisFile(File file) throws IOException
129    {
130        if (basisFile != null)
131        {
132            basisFile.close();
133            basisFile = null;
134        }
135        if (file != null)
136            basisFile = new RandomAccessFile(file, "r");
137    }
138
139    /**
140     * Set the basis file.
141     *
142     * @param file The basis file name.
143     * @throws IOException If the file name is not the name of a readable file.
144     */
145    public void setBasisFile(String file) throws IOException
146    {
147        if (basisFile != null)
148        {
149            basisFile.close();
150            basisFile = null;
151        }
152        if (file != null)
153            basisFile = new RandomAccessFile(file, "r");
154    }
155
156    /**
157     *
158     */
159    public void doFinal() throws IOException
160    {
161        if (basisFile != null)
162            basisFile.close();
163    }
164
165    /**
166     * Update this rebuilder with a delta.
167     *
168     * @param delta The delta to apply.
169     * @throws IOException If there is an error reading from the basis
170     *                     file, or if no basis file has been specified.
171     */
172    public void update(Delta delta) throws IOException, ListenerException
173    {
174        ListenerException exception = null, current = null;
175        RebuilderEvent e = null;
176        if (delta instanceof DataBlock)
177        {
178            e = new RebuilderEvent(((DataBlock) delta).getData(),
179                    delta.getWriteOffset());
180        } else
181        {
182            if (basisFile == null)
183                throw new IOException("offsets found but no basis file specified");
184            int len = Math.min(delta.getBlockLength(),
185                    (int) (basisFile.length() - ((Offsets) delta).getOldOffset()));
186            if (len < 0)
187                return;
188            byte[] buf = new byte[len];
189            basisFile.seek(((Offsets) delta).getOldOffset());
190            len = basisFile.read(buf);
191            e = new RebuilderEvent(buf, 0, len, delta.getWriteOffset());
192        }
193        for (RebuilderListener listener : listeners)
194        {
195            try
196            {
197                (listener).update(e);
198            } catch (ListenerException le)
199            {
200                if (exception != null)
201                {
202                    current.setNext(le);
203                    current = le;
204                } else
205                {
206                    exception = le;
207                    current = le;
208                }
209            }
210        }
211        if (exception != null)
212            throw exception;
213    }
214}

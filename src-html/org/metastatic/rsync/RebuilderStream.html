024package org.metastatic.rsync;
025
026import java.io.File;
027import java.io.IOException;
028import java.io.RandomAccessFile;
029import java.util.LinkedList;
030
031/**
032 * A "streaming" alternative to {@link Rebuilder}. To use this class,
033 * create an intsance with a file argument representing the file being
034 * rebuilt. Then register one or more implementations of the {@link
035 * RebuilderListener} interface, which will write the data to the new
036 * file. Then call the {@link #update(Delta)} method for each {@link
037 * Delta} to be applied.
038 * <p/>
039 * <p>Note that unlike the {@link GeneratorStream} and {@link
040 * MatcherStream} classes this class does not need a {@link
041 * Configuration}, nor does it have any "doFinal" method -- it is
042 * completely stateless (except for the file) and the operations are
043 * finished when the last delta has been applied.
044 * <p/>
045 * <p>This class is optimal for situations where the deltas are coming
046 * in a stream over a communications link, and when it would be
047 * inefficient to wait until all deltas are received.
048 */
049public class RebuilderStream
050{
051
052    // Fields.
053    // -----------------------------------------------------------------------
054
055    /**
056     * The basis file.
057     */
058    protected RandomAccessFile basisFile;
059
060    /**
061     * The list of {@link RebuilderListener}s.
062     */
063    protected final LinkedList<RebuilderListener> listeners;
064
065    // Constructors.
066    // -----------------------------------------------------------------------
067
068    /**
069     * Create a new rebuilder.
070     */
071    public RebuilderStream()
072    {
073        listeners = new LinkedList<RebuilderListener>();
074    }
075
076    // Instance methods.
077    // -----------------------------------------------------------------------
078
079    /**
080     * Add a RebuilderListener listener to this rebuilder.
081     *
082     * @param listener The listener to add.
083     * @throws IllegalArgumentException If <i>listener</i> is null.
084     */
085    public void addListener(RebuilderListener listener)
086    {
087        if (listener == null)
088            throw new IllegalArgumentException();
089        listeners.add(listener);
090    }
091
092    /**
093     * Remove a listener from this rebuilder.
094     *
095     * @param listener The listener to remove.
096     */
097    public void removeListener(RebuilderListener listener)
098    {
099        listeners.remove(listener);
100    }
101
102    /**
103     * Set the basis file.
104     *
105     * @param file The basis file.
106     * @throws IOException If the file is not readable.
107     */
108    public void setBasisFile(File file) throws IOException
109    {
110        if (basisFile != null)
111        {
112            basisFile.close();
113            basisFile = null;
114        }
115        if (file != null)
116            basisFile = new RandomAccessFile(file, "r");
117    }
118
119    /**
120     * Set the basis file.
121     *
122     * @param file The basis file name.
123     * @throws IOException If the file name is not the name of a readable file.
124     */
125    public void setBasisFile(String file) throws IOException
126    {
127        if (basisFile != null)
128        {
129            basisFile.close();
130            basisFile = null;
131        }
132        if (file != null)
133            basisFile = new RandomAccessFile(file, "r");
134    }
135
136    /**
137     *
138     */
139    public void doFinal() throws IOException
140    {
141        if (basisFile != null)
142            basisFile.close();
143    }
144
145    /**
146     * Update this rebuilder with a delta.
147     *
148     * @param delta The delta to apply.
149     * @throws IOException If there is an error reading from the basis
150     *                     file, or if no basis file has been specified.
151     */
152    public void update(Delta delta) throws IOException, ListenerException
153    {
154        ListenerException exception = null, current = null;
155        RebuilderEvent e = null;
156        if (delta instanceof DataBlock)
157        {
158            e = new RebuilderEvent(((DataBlock) delta).getData(),
159                    delta.getWriteOffset());
160        } else
161        {
162            if (basisFile == null)
163                throw new IOException("offsets found but no basis file specified");
164            int len = Math.min(delta.getBlockLength(),
165                    (int) (basisFile.length() - ((Offsets) delta).getOldOffset()));
166            if (len < 0)
167                return;
168            byte[] buf = new byte[len];
169            basisFile.seek(((Offsets) delta).getOldOffset());
170            len = basisFile.read(buf);
171            e = new RebuilderEvent(buf, 0, len, delta.getWriteOffset());
172        }
173        for (RebuilderListener listener : listeners)
174        {
175            try
176            {
177                (listener).update(e);
178            } catch (ListenerException le)
179            {
180                if (exception != null)
181                {
182                    current.setNext(le);
183                    current = le;
184                } else
185                {
186                    exception = le;
187                    current = le;
188                }
189            }
190        }
191        if (exception != null)
192            throw exception;
193    }
194}

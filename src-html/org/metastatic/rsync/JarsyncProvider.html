023package org.metastatic.rsync;
024
025/**
026 * This provider implements the MD4 message digest, and is provided to
027 * ensure that MD4 is available.
028 */
029public final class JarsyncProvider extends java.security.Provider
030{
031    public JarsyncProvider()
032    {
033        super("JARSYNC", org.metastatic.rsync.version.VERSION_DOUBLE,
034                "Jarsync provider; implementing MD4, BrokenMD4, xxhash64");
035
036        put("MessageDigest.MD4", "org.metastatic.rsync.MD4");
037        put("MessageDigest.BrokenMD4", "org.metastatic.rsync.BrokenMD4");
038        put("MessageDigest.Murmur3", "org.metastatic.rsync.Murmur3Adapter");
039    }
040}

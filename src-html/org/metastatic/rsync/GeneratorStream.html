042package org.metastatic.rsync;
043
044import java.util.LinkedList;
045import java.util.List;
046
047/**
048 * A "streaming" API for generating checksums. This class accepts incremental
049 * byte values of any size as input, and sends {@link org.metastatic.rsync.GeneratorEvent}
050 * objects to listeners as checksums are generated.
051 *
052 * <p>The general usage pattern:</p>
053 *
054 * <pre>GeneratorStream generator = new GeneratorStream(config);
055generator.addListener(your_listener);
056for (byte[] part : parts)
057   generator.update(part);
058generator.doFinal();</pre>
059 */
060public class GeneratorStream
061{
062
063    // Constants and fields.
064    // -----------------------------------------------------------------------
065
066    /**
067     * The configuration.
068     */
069    protected final Configuration config;
070
071    /**
072     * The list of {@link GeneratorListener}s.
073     */
074    protected final List<GeneratorListener> listeners;
075
076    /**
077     * The intermediate byte buffer.
078     */
079    protected final byte[] buffer;
080
081    /**
082     * The current index in {@link #buffer}.
083     */
084    protected int ndx;
085
086    /**
087     * The number of bytes summed thusfar.
088     */
089    protected long count;
090
091    // Constructor.
092    // -----------------------------------------------------------------------
093
094    public GeneratorStream(Configuration config)
095    {
096        this.config = config;
097        this.listeners = new LinkedList<GeneratorListener>();
098        buffer = new byte[config.blockLength];
099        reset();
100    }
101
102    // Instance methods.
103    // -----------------------------------------------------------------------
104
105    /**
106     * Add a {@link GeneratorListener} to the list of listeners.
107     *
108     * @param listener The listener to add.
109     */
110    public void addListener(GeneratorListener listener)
111    {
112        if (listener == null)
113            throw new IllegalArgumentException();
114        listeners.add(listener);
115    }
116
117    /**
118     * Remove a {@link GeneratorListener} from the list of listeners.
119     *
120     * @param listener The listener to add.
121     * @return True if a listener was really removed (i.e. that the
122     * listener was in the list to begin with).
123     */
124    public boolean removeListener(GeneratorListener listener)
125    {
126        return listeners.remove(listener);
127    }
128
129    /**
130     * Reset this generator, to be used for another data set.
131     */
132    public void reset()
133    {
134        ndx = 0;
135        count = 0L;
136    }
137
138    /**
139     * Update this generator with a single byte.
140     *
141     * @param b The next byte
142     */
143    public void update(byte b) throws ListenerException
144    {
145        ListenerException exception = null, current = null;
146        buffer[ndx++] = b;
147        if (ndx == buffer.length)
148        {
149            ChecksumPair p = generateSum(buffer, 0, buffer.length);
150            for (GeneratorListener listener : listeners)
151            {
152                try
153                {
154                    listener.update(new GeneratorEvent(p));
155                } catch (ListenerException le)
156                {
157                    if (exception != null)
158                    {
159                        current.setNext(le);
160                        current = le;
161                    } else
162                    {
163                        exception = le;
164                        current = le;
165                    }
166                }
167            }
168            if (exception != null)
169                throw exception;
170            ndx = 0;
171        }
172    }
173
174    /**
175     * Update this generator with a portion of a byte array.
176     *
177     * @param buf The next bytes.
178     * @param off The offset to begin at.
179     * @param len The number of bytes to update.
180     */
181    public void update(byte[] buf, int off, int len) throws ListenerException
182    {
183        ListenerException exception = null, current = null;
184        int i = off;
185        do
186        {
187            int l = Math.min(len - (i - off), buffer.length - ndx);
188            System.arraycopy(buf, i, buffer, ndx, l);
189            i += l;
190            ndx += l;
191            if (ndx == buffer.length)
192            {
193                ChecksumPair p = generateSum(buffer, 0, buffer.length);
194                for (GeneratorListener listener : listeners)
195                {
196                    try
197                    {
198                        listener.update(new GeneratorEvent(p));
199                    } catch (ListenerException le)
200                    {
201                        if (exception != null)
202                        {
203                            current.setNext(le);
204                            current = le;
205                        } else
206                        {
207                            exception = le;
208                            current = le;
209                        }
210                    }
211                }
212                if (exception != null)
213                    throw exception;
214                ndx = 0;
215            }
216        } while (i < off + len);
217    }
218
219    /**
220     * Update this generator with a byte array.
221     *
222     * @param buf The next bytes.
223     */
224    public void update(byte[] buf) throws ListenerException
225    {
226        update(buf, 0, buf.length);
227    }
228
229    /**
230     * Finish generating checksums, flushing any buffered data and
231     * resetting this instance.
232     */
233    public void doFinal() throws ListenerException
234    {
235        ListenerException exception = null, current = null;
236        if (ndx > 0)
237        {
238            ChecksumPair p = generateSum(buffer, 0, ndx);
239            for (GeneratorListener listener : listeners)
240            {
241                try
242                {
243                    listener.update(new GeneratorEvent(p));
244                } catch (ListenerException le)
245                {
246                    if (exception != null)
247                    {
248                        current.setNext(le);
249                        current = le;
250                    } else
251                    {
252                        exception = le;
253                        current = le;
254                    }
255                }
256            }
257            if (exception != null)
258                throw exception;
259        }
260        reset();
261    }
262
263    // Own methods.
264    // -----------------------------------------------------------------------
265
266    /**
267     * Generate a sum pair for a portion of a byte array.
268     *
269     * @param buf The byte array to checksum.
270     * @param off Where in <code>buf</code> to start.
271     * @param len How many bytes to checksum.
272     * @return A {@link ChecksumPair} for this byte array.
273     */
274    protected ChecksumPair generateSum(byte[] buf, int off, int len)
275    {
276        ChecksumPair p = new ChecksumPair();
277        config.weakSum.check(buf, off, len);
278        config.strongSum.update(buf, off, len);
279        if (config.checksumSeed != null)
280        {
281            config.strongSum.update(config.checksumSeed, 0,
282                    config.checksumSeed.length);
283        }
284        p.weak = config.weakSum.getValue();
285        p.strong = new byte[config.strongSumLength];
286        System.arraycopy(config.strongSum.digest(), 0, p.strong, 0,
287                config.strongSumLength);
288        p.offset = count;
289        p.length = len;
290        count += len;
291        return p;
292    }
293}

024package org.metastatic.rsync;
025
026import java.util.LinkedList;
027import java.util.List;
028
029/**
030 * A "streaming" API for generating checksums. This class accepts incremental
031 * byte values of any size as input, and sends {@link org.metastatic.rsync.GeneratorEvent}
032 * objects to listeners as checksums are generated.
033 *
034 * <p>The general usage pattern:</p>
035 *
036 * <pre>GeneratorStream generator = new GeneratorStream(config);
037generator.addListener(your_listener);
038for (byte[] part : parts)
039   generator.update(part);
040generator.doFinal();</pre>
041 */
042public class GeneratorStream
043{
044
045    // Constants and fields.
046    // -----------------------------------------------------------------------
047
048    /**
049     * The configuration.
050     */
051    protected final Configuration config;
052
053    /**
054     * The list of {@link GeneratorListener}s.
055     */
056    protected final List<GeneratorListener> listeners;
057
058    /**
059     * The intermediate byte buffer.
060     */
061    protected final byte[] buffer;
062
063    /**
064     * The current index in {@link #buffer}.
065     */
066    protected int ndx;
067
068    /**
069     * The number of bytes summed thusfar.
070     */
071    protected long count;
072
073    protected int seq;
074
075    // Constructor.
076    // -----------------------------------------------------------------------
077
078    public GeneratorStream(Configuration config)
079    {
080        this.config = config;
081        this.listeners = new LinkedList<GeneratorListener>();
082        buffer = new byte[config.blockLength];
083        reset();
084    }
085
086    // Instance methods.
087    // -----------------------------------------------------------------------
088
089    /**
090     * Add a {@link GeneratorListener} to the list of listeners.
091     *
092     * @param listener The listener to add.
093     */
094    public void addListener(GeneratorListener listener)
095    {
096        if (listener == null)
097            throw new IllegalArgumentException();
098        listeners.add(listener);
099    }
100
101    /**
102     * Remove a {@link GeneratorListener} from the list of listeners.
103     *
104     * @param listener The listener to add.
105     * @return True if a listener was really removed (i.e. that the
106     * listener was in the list to begin with).
107     */
108    public boolean removeListener(GeneratorListener listener)
109    {
110        return listeners.remove(listener);
111    }
112
113    /**
114     * Reset this generator, to be used for another data set.
115     */
116    public void reset()
117    {
118        ndx = 0;
119        count = 0L;
120        seq = 0;
121    }
122
123    /**
124     * Update this generator with a single byte.
125     *
126     * @param b The next byte
127     */
128    public void update(byte b) throws ListenerException
129    {
130        ListenerException exception = null, current = null;
131        buffer[ndx++] = b;
132        if (ndx == buffer.length)
133        {
134            ChecksumLocation p = generateSum(buffer, 0, buffer.length);
135            for (GeneratorListener listener : listeners)
136            {
137                try
138                {
139                    listener.update(new GeneratorEvent(p));
140                } catch (ListenerException le)
141                {
142                    if (exception != null)
143                    {
144                        current.setNext(le);
145                        current = le;
146                    } else
147                    {
148                        exception = le;
149                        current = le;
150                    }
151                }
152            }
153            if (exception != null)
154                throw exception;
155            ndx = 0;
156        }
157    }
158
159    /**
160     * Update this generator with a portion of a byte array.
161     *
162     * @param buf The next bytes.
163     * @param off The offset to begin at.
164     * @param len The number of bytes to update.
165     */
166    public void update(byte[] buf, int off, int len) throws ListenerException
167    {
168        ListenerException exception = null, current = null;
169        int i = off;
170        do
171        {
172            int l = Math.min(len - (i - off), buffer.length - ndx);
173            System.arraycopy(buf, i, buffer, ndx, l);
174            i += l;
175            ndx += l;
176            if (ndx == buffer.length)
177            {
178                ChecksumLocation p = generateSum(buffer, 0, buffer.length);
179                for (GeneratorListener listener : listeners)
180                {
181                    try
182                    {
183                        listener.update(new GeneratorEvent(p));
184                    } catch (ListenerException le)
185                    {
186                        if (exception != null)
187                        {
188                            current.setNext(le);
189                            current = le;
190                        } else
191                        {
192                            exception = le;
193                            current = le;
194                        }
195                    }
196                }
197                if (exception != null)
198                    throw exception;
199                ndx = 0;
200            }
201        } while (i < off + len);
202    }
203
204    /**
205     * Update this generator with a byte array.
206     *
207     * @param buf The next bytes.
208     */
209    public void update(byte[] buf) throws ListenerException
210    {
211        update(buf, 0, buf.length);
212    }
213
214    /**
215     * Finish generating checksums, flushing any buffered data and
216     * resetting this instance.
217     */
218    public void doFinal() throws ListenerException
219    {
220        ListenerException exception = null, current = null;
221        if (ndx > 0)
222        {
223            ChecksumLocation p = generateSum(buffer, 0, ndx);
224            for (GeneratorListener listener : listeners)
225            {
226                try
227                {
228                    listener.update(new GeneratorEvent(p));
229                } catch (ListenerException le)
230                {
231                    if (exception != null)
232                    {
233                        current.setNext(le);
234                        current = le;
235                    } else
236                    {
237                        exception = le;
238                        current = le;
239                    }
240                }
241            }
242            if (exception != null)
243                throw exception;
244        }
245        reset();
246    }
247
248    // Own methods.
249    // -----------------------------------------------------------------------
250
251    /**
252     * Generate a sum pair for a portion of a byte array.
253     *
254     * @param buf The byte array to checksum.
255     * @param off Where in <code>buf</code> to start.
256     * @param len How many bytes to checksum.
257     * @return A {@link org.metastatic.rsync.ChecksumLocation} for this byte array.
258     */
259    protected ChecksumLocation generateSum(byte[] buf, int off, int len)
260    {
261        config.weakSum.check(buf, off, len);
262        if (config.checksumSeed != null && config.isSeedPrefix)
263            config.strongSum.update(config.checksumSeed);
264        config.strongSum.update(buf, off, len);
265        if (config.checksumSeed != null && !config.isSeedPrefix)
266            config.strongSum.update(config.checksumSeed);
267        int weak = config.weakSum.getValue();
268        byte[] strong = new byte[config.strongSumLength];
269        System.arraycopy(config.strongSum.digest(), 0, strong, 0, config.strongSumLength);
270        ChecksumLocation loc = new ChecksumLocation(new ChecksumPair(weak, strong), count, len, seq);
271        count += len;
272        seq++;
273        return loc;
274    }
275}

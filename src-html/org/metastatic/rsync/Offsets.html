023package org.metastatic.rsync;
024
025/**
026 * This class represents an update to a file or array of bytes wherein
027 * the bytes themselves have not changed, but have moved to another
028 * location. This is represented by three fields: the offset in the
029 * original data, the offset in the new data, and the length, in bytes,
030 * of this block.
031 *
032 * @version $Revision$
033 */
034public class Offsets implements Delta, java.io.Serializable
035{
036
037    // Constants and variables
038    // ------------------------------------------------------------------------
039
040    private static final long serialVersionUID = 2787420454508237262L;
041
042    /**
043     * The original offset.
044     *
045     * @since 1.1
046     */
047    protected long oldOffset;
048
049    /**
050     * The new offset.
051     *
052     * @since 1.1
053     */
054    protected long newOffset;
055
056    /**
057     * The size of the moved block, in bytes.
058     *
059     * @since 1.1
060     */
061    protected int blockLength;
062
063    // Constructors
064    // -----------------------------------------------------------------
065
066    /**
067     * Create a new pair of offsets. The idea behind this object is
068     * that this sort of {@link Delta} represents original data
069     * that has simply moved in the new data.
070     *
071     * @param oldOffset   The offset in the original data.
072     * @param newOffset   The offset in the new data.
073     * @param blockLength The size, in bytes, of the block that has moved.
074     * @since 1.1
075     */
076    public Offsets(long oldOffset, long newOffset, int blockLength)
077    {
078        this.oldOffset = oldOffset;
079        this.newOffset = newOffset;
080        this.blockLength = blockLength;
081    }
082
083    // Instance methods.
084    // -----------------------------------------------------------------------
085
086    // Delta interface implementation.
087
088    public long getWriteOffset()
089    {
090        return newOffset;
091    }
092
093    public int getBlockLength()
094    {
095        return blockLength;
096    }
097
098    // Property accessor methods
099
100    /**
101     * Get the original offset.
102     *
103     * @return The original offset.
104     */
105    public long getOldOffset()
106    {
107        return oldOffset;
108    }
109
110    /**
111     * Set the original offset.
112     *
113     * @param off The new value for the original offset.
114     */
115    public void setOldOffset(long off)
116    {
117        oldOffset = off;
118    }
119
120    /**
121     * Get the updated offset.
122     *
123     * @return The updated offset.
124     */
125    public long getNewOffset()
126    {
127        return newOffset;
128    }
129
130    /**
131     * Set the updated offset.
132     *
133     * @param off The new value for the updated offset.
134     */
135    public void setNewOffset(long off)
136    {
137        newOffset = off;
138    }
139
140    /**
141     * Set the block size.
142     *
143     * @param len The new value for the block size.
144     */
145    public void setBlockLength(int len)
146    {
147        blockLength = len;
148    }
149
150    // Public instance methods overriding java.lang.Object -------------
151
152    /**
153     * Return a {@link java.lang.String} representation of this object.
154     *
155     * @return A string representing this object.
156     */
157    public String toString()
158    {
159        return "Offsets(old=" + oldOffset + " new=" + newOffset
160                + " len=" + blockLength + ")";
161    }
162
163    /**
164     * Test if one object is equal to this one.
165     *
166     * @return <tt>true</tt> If <tt>o</tt> is an Offsets instance and the
167     * {@link #oldOffset}, {@link #newOffset}, and {@link
168     * #blockLength} fields are all equal.
169     * @throws java.lang.ClassCastException   If <tt>o</tt> is not an
170     *                                        instance of this class.
171     * @throws java.lang.NullPointerException If <tt>o</tt> is null.
172     */
173    public boolean equals(Object o)
174    {
175        return oldOffset == ((Offsets) o).oldOffset
176                && newOffset == ((Offsets) o).newOffset
177                && blockLength == ((Offsets) o).blockLength;
178    }
179
180    /**
181     * Returns the hash code of this object, defined as:
182     * <blockquote>
183     * <tt>{@link #oldOffset} + {@link #newOffset} + {@link
184     * #blockLength}
185     * % 2^32</tt>
186     * </blockquote>
187     *
188     * @return The hash code of this object.
189     */
190    public int hashCode()
191    {
192        return (int) (oldOffset + newOffset + blockLength);
193    }
194}

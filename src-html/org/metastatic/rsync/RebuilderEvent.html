043package org.metastatic.rsync;
044
045/**
046 * a rebuilder event. Rebuilder events are emitted by a {@link
047 * RebuilderStream} each time a new {@link Delta} is applied. The stream
048 * will send this event to each of its {@link RebuilderListener}s.
049 *
050 * @see RebuilderStream
051 * @see RebuilderListener
052 */
053public class RebuilderEvent extends java.util.EventObject
054{
055
056    // Fields.
057    // -------------------------------------------------------------------------
058
059    /**
060     * The destination offset.
061     */
062    protected transient long offset;
063
064    // Constructors.
065    // -------------------------------------------------------------------------
066
067    /**
068     * Create a new rebuilder event.
069     *
070     * @param data   The source of this event, the data block.
071     * @param offset The destination offset.
072     */
073    public RebuilderEvent(byte[] data, long offset)
074    {
075        this(data, 0, data.length, offset);
076    }
077
078    public RebuilderEvent(byte[] data, int off, int len, long offset)
079    {
080        super(new byte[len]);
081        System.arraycopy(data, off, source, 0, len);
082        this.offset = offset;
083    }
084
085    // Instance methods.
086    // -------------------------------------------------------------------------
087
088    /**
089     * Get the data. This method is equivalent to {@link
090     * java.util.EventObject#getSource()} but the source is already cast
091     * for convenience.
092     *
093     * @return The data array.
094     */
095    public byte[] getData()
096    {
097        return (byte[]) source;
098    }
099
100    /**
101     * Get the offset at which the data should be written.
102     *
103     * @return The offset.
104     */
105    public long getOffset()
106    {
107        return offset;
108    }
109}

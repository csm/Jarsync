024package org.metastatic.rsync;
025
026/**
027 * a rebuilder event. Rebuilder events are emitted by a {@link
028 * RebuilderStream} each time a new {@link Delta} is applied. The stream
029 * will send this event to each of its {@link RebuilderListener}s.
030 *
031 * @see RebuilderStream
032 * @see RebuilderListener
033 */
034public class RebuilderEvent extends java.util.EventObject
035{
036
037    // Fields.
038    // -------------------------------------------------------------------------
039
040    /**
041     * The destination offset.
042     */
043    protected transient long offset;
044
045    // Constructors.
046    // -------------------------------------------------------------------------
047
048    /**
049     * Create a new rebuilder event.
050     *
051     * @param data   The source of this event, the data block.
052     * @param offset The destination offset.
053     */
054    public RebuilderEvent(byte[] data, long offset)
055    {
056        this(data, 0, data.length, offset);
057    }
058
059    public RebuilderEvent(byte[] data, int off, int len, long offset)
060    {
061        super(new byte[len]);
062        System.arraycopy(data, off, source, 0, len);
063        this.offset = offset;
064    }
065
066    // Instance methods.
067    // -------------------------------------------------------------------------
068
069    /**
070     * Get the data. This method is equivalent to {@link
071     * java.util.EventObject#getSource()} but the source is already cast
072     * for convenience.
073     *
074     * @return The data array.
075     */
076    public byte[] getData()
077    {
078        return (byte[]) source;
079    }
080
081    /**
082     * Get the offset at which the data should be written.
083     *
084     * @return The offset.
085     */
086    public long getOffset()
087    {
088        return offset;
089    }
090}

038package org.metastatic.rsync;
039
040import com.google.common.hash.HashFunction;
041import com.google.common.hash.Hasher;
042import com.google.common.hash.Hashing;
043
044import java.security.MessageDigestSpi;
045
046public class Murmur3Adapter extends MessageDigestSpi
047{
048    private final HashFunction hashFunction;
049    private Hasher hasher;
050
051    public Murmur3Adapter()
052    {
053        hashFunction = Hashing.murmur3_128();
054        hasher = hashFunction.newHasher();
055    }
056
057    @Override
058    protected int engineGetDigestLength()
059    {
060        return hashFunction.bits() / 8;
061    }
062
063    @Override
064    protected void engineUpdate(byte input)
065    {
066        hasher.putByte(input);
067    }
068
069    @Override
070    protected void engineUpdate(byte[] input, int offset, int len)
071    {
072        hasher.putBytes(input, offset, len);
073    }
074
075    @Override
076    protected byte[] engineDigest()
077    {
078        byte[] ret = hasher.hash().asBytes();
079        engineReset();
080        return ret;
081    }
082
083    @Override
084    protected void engineReset()
085    {
086        hasher = hashFunction.newHasher();
087    }
088}

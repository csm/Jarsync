044package org.metastatic.rsync;
045
046import java.io.*;
047
048import java.util.LinkedList;
049import java.util.List;
050
051/**
052 * <p>Methods for performing the checksum search. The result of a search
053 * is a {@link java.util.List} of {@link Delta} objects that, when
054 * applied to a method in {@link Rebuilder}, will reconstruct the new
055 * version of the data.</p>
056 *
057 * @version $Revision$
058 */
059public final class Matcher {
060
061   // Constants and variables.
062   // -----------------------------------------------------------------
063
064   /** The list of deltas being built. */
065   protected final List<Delta> deltas;
066
067   /** The underlying matcher stream. */
068   protected final MatcherStream matcher;
069
070   /** The size of allocated byte arrays. */
071   protected final int chunkSize;
072
073   // Constructors.
074   // -----------------------------------------------------------------
075
076   /**
077    * Create a matcher with the specified configuration.
078    *
079    * @param config The {@link Configuration} for this Matcher.
080    */
081   public Matcher(Configuration config) {
082      deltas = new LinkedList<Delta>();
083      matcher = new MatcherStream(config);
084      matcher.addListener(new Callback(deltas));
085      chunkSize = config.chunkSize;
086   }
087
088 // Instance methods.
089   // -----------------------------------------------------------------
090
091   /**
092    * Search the given byte buffer.
093    *
094    * @param sums The checksums to search for.
095    * @param buf  The data buffer to search.
096    * @return A collection of {@link Delta}s derived from this search.
097    */
098   public List<Delta> hashSearch(List<ChecksumPair> sums, byte[] buf) {
099      return hashSearch(sums, buf, 0, buf.length);
100   }
101
102   /**
103    * Search a portion of a byte buffer.
104    *
105    * @param sums The checksums to search for.
106    * @param buf  The data buffer to search.
107    * @param off  The offset in <code>buf</code> to begin.
108    * @param len  The number of bytes to search from <code>buf</code>.
109    * @return A collection of {@link Delta}s derived from this search.
110    */
111   public List<Delta> hashSearch(List<ChecksumPair> sums, byte[] buf, int off, int len) {
112      deltas.clear();
113      matcher.reset();
114      matcher.setChecksums(sums);
115      try {
116         matcher.update(buf, off, len);
117         matcher.doFinal();
118      } catch (ListenerException shouldNotHappen) {
119      }
120      return new LinkedList<Delta>(deltas);
121   }
122
123   /**
124    * Search a file by name.
125    *
126    * @param sums The checksums to search for.
127    * @param filename The name of the file to search.
128    * @return A list of deltas derived from this search.
129    * @throws IOException If <i>filename</i> cannot be read.
130    */
131   public List<Delta> hashSearch(List sums, String filename) throws IOException {
132      return hashSearch(sums, new FileInputStream(filename));
133   }
134
135   /**
136    * Search a file.
137    *
138    * @param sums The checksums to search for.
139    * @param f    The file to search.
140    * @return A list of {@link Delta}s derived from this search.
141    * @throws IOException If <i>f</i> cannot be read.
142    */
143   public List<Delta> hashSearch(List<ChecksumPair> sums, File f) throws IOException {
144      return hashSearch(sums, new FileInputStream(f));
145   }
146
147   /**
148    * Search an input stream.
149    *
150    * @param sums  The checksums to search.
151    * @param in The input stream to search.
152    * @return A collection of {@link Delta}s derived from this search.
153    * @throws IOException If an exception occurs while reading.
154    */
155   public List<Delta> hashSearch(List<ChecksumPair> sums, InputStream in) throws IOException {
156      deltas.clear();
157      matcher.reset();
158      matcher.setChecksums(sums);
159      byte[] buffer = new byte[chunkSize];
160      int len = 0;
161      try {
162         while ((len = in.read(buffer)) != -1)
163            matcher.update(buffer, 0, len);
164         matcher.doFinal();
165      } catch (ListenerException shouldNeverHappen) {
166      }
167      return new LinkedList<Delta>(deltas);
168   }
169
170   // Inner classes.
171   // -----------------------------------------------------------------------
172
173   /**
174    * Trivial implementation of a MatcherListener that simply adds
175    * incoming deltas to a List.
176    */
177   private class Callback implements MatcherListener {
178
179      // Fields.
180      // --------------------------------------------------------------------
181
182      private final List<Delta> deltas;
183
184      // Constructors.
185      // --------------------------------------------------------------------
186
187      Callback(List<Delta> deltas) {
188         this.deltas = deltas;
189      }
190
191      // Instance methods.
192      // --------------------------------------------------------------------
193
194      public void update(MatcherEvent event) {
195         deltas.add(event.getDelta());
196      }
197   }
198}

023package org.metastatic.rsync;
024
025import java.io.*;
026
027import java.util.LinkedList;
028import java.util.List;
029
030/**
031 * <p>Methods for performing the checksum search. The result of a search
032 * is a {@link java.util.List} of {@link Delta} objects that, when
033 * applied to a method in {@link Rebuilder}, will reconstruct the new
034 * version of the data.</p>
035 *
036 * @version $Revision$
037 */
038public final class Matcher {
039
040   // Constants and variables.
041   // -----------------------------------------------------------------
042
043   /** The list of deltas being built. */
044   protected final List<Delta> deltas;
045
046   /** The underlying matcher stream. */
047   protected final MatcherStream matcher;
048
049   /** The size of allocated byte arrays. */
050   protected final int chunkSize;
051
052   // Constructors.
053   // -----------------------------------------------------------------
054
055   /**
056    * Create a matcher with the specified configuration.
057    *
058    * @param config The {@link Configuration} for this Matcher.
059    */
060   public Matcher(Configuration config) {
061      deltas = new LinkedList<Delta>();
062      matcher = new MatcherStream(config);
063      matcher.addListener(new Callback(deltas));
064      chunkSize = config.chunkSize;
065   }
066
067 // Instance methods.
068   // -----------------------------------------------------------------
069
070   /**
071    * Search the given byte buffer.
072    *
073    * @param sums The checksums to search for.
074    * @param buf  The data buffer to search.
075    * @return A collection of {@link Delta}s derived from this search.
076    */
077   public List<Delta> hashSearch(List<ChecksumLocation> sums, byte[] buf) {
078      return hashSearch(sums, buf, 0, buf.length);
079   }
080
081   /**
082    * Search a portion of a byte buffer.
083    *
084    * @param sums The checksums to search for.
085    * @param buf  The data buffer to search.
086    * @param off  The offset in <code>buf</code> to begin.
087    * @param len  The number of bytes to search from <code>buf</code>.
088    * @return A collection of {@link Delta}s derived from this search.
089    */
090   public List<Delta> hashSearch(List<ChecksumLocation> sums, byte[] buf, int off, int len)
091   {
092      deltas.clear();
093      matcher.reset();
094      matcher.setChecksums(sums);
095      try {
096         matcher.update(buf, off, len);
097         matcher.doFinal();
098      } catch (ListenerException shouldNotHappen) {
099      }
100      return new LinkedList<Delta>(deltas);
101   }
102
103   /**
104    * Search a file by name.
105    *
106    * @param sums The checksums to search for.
107    * @param filename The name of the file to search.
108    * @return A list of deltas derived from this search.
109    * @throws IOException If <i>filename</i> cannot be read.
110    */
111   public List<Delta> hashSearch(List<ChecksumLocation> sums, String filename) throws IOException {
112      return hashSearch(sums, new FileInputStream(filename));
113   }
114
115   /**
116    * Search a file.
117    *
118    * @param sums The checksums to search for.
119    * @param f    The file to search.
120    * @return A list of {@link Delta}s derived from this search.
121    * @throws IOException If <i>f</i> cannot be read.
122    */
123   public List<Delta> hashSearch(List<ChecksumLocation> sums, File f) throws IOException {
124      return hashSearch(sums, new FileInputStream(f));
125   }
126
127   /**
128    * Search an input stream.
129    *
130    * @param sums  The checksums to search.
131    * @param in The input stream to search.
132    * @return A collection of {@link Delta}s derived from this search.
133    * @throws IOException If an exception occurs while reading.
134    */
135   public List<Delta> hashSearch(List<ChecksumLocation> sums, InputStream in) throws IOException {
136      deltas.clear();
137      matcher.reset();
138      matcher.setChecksums(sums);
139      byte[] buffer = new byte[chunkSize];
140      int len = 0;
141      try {
142         while ((len = in.read(buffer)) != -1)
143            matcher.update(buffer, 0, len);
144         matcher.doFinal();
145      } catch (ListenerException shouldNeverHappen) {
146      }
147      return new LinkedList<Delta>(deltas);
148   }
149
150   // Inner classes.
151   // -----------------------------------------------------------------------
152
153   /**
154    * Trivial implementation of a MatcherListener that simply adds
155    * incoming deltas to a List.
156    */
157   private class Callback implements MatcherListener {
158
159      // Fields.
160      // --------------------------------------------------------------------
161
162      private final List<Delta> deltas;
163
164      // Constructors.
165      // --------------------------------------------------------------------
166
167      Callback(List<Delta> deltas) {
168         this.deltas = deltas;
169      }
170
171      // Instance methods.
172      // --------------------------------------------------------------------
173
174      public void update(MatcherEvent event) {
175         deltas.add(event.getDelta());
176      }
177   }
178}
